33 template <typename ScalarH2, typename DenseRhsType, typename DenseResType,
34  typename AlphaType>
35 struct H2_time_dense_product_impl<H2Matrix<ScalarH2>, DenseRhsType,
36  DenseResType, AlphaType, ColMajor, true> {
37  typedef typename traits<DenseRhsType>::Scalar ScalarRhs;
38  typedef typename traits<DenseResType>::Scalar ScalarRes;
39  static void run(const H2Matrix<ScalarH2>& lhs, const DenseRhsType& rhs,
40  DenseResType& res, const AlphaType& alpha) {
41  // get H2-data
42  int max_level =
43  lhs.get_block_cluster_tree()(0, 0).get_parameters().max_level_;
44  int min_cluster_level =
45  lhs.get_block_cluster_tree()(0, 0).get_parameters().min_cluster_level_;
46  auto moment_matrix = lhs.get_fmm_moment_matrix();
47  auto transfer_matrices = lhs.get_fmm_transfer_matrices();
48  int vector_dimension = moment_matrix.size();
49 
50  for (Index c = 0; c < rhs.cols(); ++c) {
51  // go discontinuous in rhs
52  Matrix<ScalarH2, Dynamic, 1> long_rhs_all =
53  (lhs.get_transformation_matrix() * rhs.col(c)).eval();
54  int vector_component_size = long_rhs_all.rows() / vector_dimension;
55 
56  // initialize destination
57  Matrix<ScalarRes, Dynamic, 1> long_dst_all(long_rhs_all.rows());
58  long_dst_all.setZero();
59 
60  for (int col_component = 0; col_component < vector_dimension;
61  ++col_component) {
62  for (int row_component = 0; row_component < vector_dimension;
63  ++row_component) {
64  Matrix<ScalarRhs, Dynamic, 1> long_rhs = long_rhs_all.segment(
65  col_component * vector_component_size, vector_component_size);
66  Matrix<ScalarRhs, Dynamic, 1> long_dst(long_rhs.rows());
67  long_dst.setZero();
68 
69  // split long rhs into pieces by reshaping
70  Matrix<ScalarRhs, Dynamic, Dynamic> long_rhs_matrix =
71  Map<Matrix<ScalarRhs, Dynamic, Dynamic>>(
72  long_rhs.data(), moment_matrix[col_component].cols(),
73  long_rhs.rows() / moment_matrix[col_component].cols());
74 
75  // do forward-transformation
76  std::vector<Matrix<ScalarRhs, Dynamic, Dynamic>> long_rhs_forward =
77  Bembel::H2Multipole::forwardTransformation(
78  moment_matrix[col_component], transfer_matrices,
79  max_level - min_cluster_level, long_rhs_matrix);
80 
81 #pragma omp parallel
82  {
83  // initialize target for each process
84  Matrix<ScalarRes, Dynamic, 1> my_long_dst(long_dst.rows());
85 
86  // initialize target of backward-transformation
87  std::vector<Matrix<ScalarRes, Dynamic, Dynamic>>
88  my_long_dst_backward;
89  for (int i = 0; i < long_rhs_forward.size(); ++i)
90  my_long_dst_backward.push_back(
91  Matrix<ScalarRes, Dynamic, Dynamic>::Zero(
92  long_rhs_forward[i].rows(), long_rhs_forward[i].cols()));
93  my_long_dst.setZero();
94 
95  // matrix-vector
96  for (auto leaf =
97  lhs.get_block_cluster_tree()(row_component, col_component)
98  .clbegin();
99  leaf !=
100  lhs.get_block_cluster_tree()(row_component, col_component)
101  .clend();
102  ++leaf) {
103 #pragma omp single nowait
104  {
105  switch ((*leaf)->get_cc()) {
106  // deal with matrix blocks
107  case Bembel::BlockClusterAdmissibility::Dense: {
108  my_long_dst.segment((*leaf)->get_row_start_index(),
109  (*leaf)->get_row_end_index() -
110  (*leaf)->get_row_start_index()) +=
111  (*leaf)->get_leaf().get_F() *
112  long_rhs.segment((*leaf)->get_col_start_index(),
113  (*leaf)->get_col_end_index() -
114  (*leaf)->get_col_start_index());
115  } break;
116  // deal with low-rank blocks
117  case Bembel::BlockClusterAdmissibility::LowRank: {
118  const Bembel::ElementTreeNode* cluster1 =
119  (*leaf)->get_cluster1();
120  const Bembel::ElementTreeNode* cluster2 =
121  (*leaf)->get_cluster2();
122  int cluster_level = cluster1->get_level();
123  int fmm_level = lhs.get_block_cluster_tree()(0, 0)
124  .get_parameters()
125  .max_level_ -
126  lhs.get_block_cluster_tree()(0, 0)
127  .get_parameters()
128  .min_cluster_level_ -
129  (*leaf)->get_cluster1()->get_level();
130  int cluster1_col = cluster1->id_;
131  int cluster2_col = cluster2->id_;
132  my_long_dst_backward[fmm_level].col(cluster1_col) +=
133  (*leaf)->get_leaf().get_F() *
134  long_rhs_forward[fmm_level].col(cluster2_col);
135  } break;
136  // this leaf is not a low-rank block and not a dense block,
137  // thus it is not a leaf -> error
138  default:
139  assert(0 && "This should never happen");
140  break;
141  }
142  }
143  }
144 
145  // do backward transformation
146  my_long_dst += Bembel::H2Multipole::backwardTransformation(
147  moment_matrix[row_component], transfer_matrices,
148  max_level - min_cluster_level, my_long_dst_backward);
149 
150 #pragma omp critical
151  long_dst += my_long_dst;
152  }
153 
154  // finish off vector component
155  long_dst_all.segment(row_component * vector_component_size,
156  vector_component_size) += long_dst;
157  }
158  }
159 
160  // go continuous and write output
161  res.col(c) +=
162  alpha * lhs.get_transformation_matrix().transpose() * long_dst_all;
163  }
164  }
165 };

196  typedef Product<BinaryRhs, Rhs, ProductType> RightProduct;
197  typedef CwiseBinaryOp<BinaryOp, LeftProduct, RightProduct> NewCwiseBinaryOp;

199  typedef CwiseNullaryOp<scalar_constant_op<Scalar>, DenseMatrix>
200  ScalarCwiseNullaryOp;
201  typedef scalar_constant_op<Scalar> ScalarConstantOp;
202  typedef scalar_product_op<Scalar, Scalar> ScalarTimesOp;
203  typedef CwiseBinaryOp<ScalarTimesOp, ScalarCwiseNullaryOp, NewCwiseBinaryOp>
204  ScalarTimesNewCwiseBinaryOp;
205  typedef typename traits<Lhs>::Scalar LhsScalar;
206  typedef typename traits<Rhs>::Scalar RhsScalar;
207  typedef add_assign_op<LhsScalar, RhsScalar> addAssignOp;
208 
209  static_assert(
210  is_same<BinaryOp, scalar_sum_op<LhsScalar, RhsScalar>>::value ||
211  is_same<BinaryOp,
212  scalar_difference_op<LhsScalar, RhsScalar>>::value,
213  "Product of CwiseBinaryOp not defined for this BinaryOp");
214 
215  LeftProduct lprod(lhs.lhs(), rhs);
216  RightProduct rprod(lhs.rhs(), rhs);
217  NewCwiseBinaryOp xpr_prod(lprod, rprod, lhs.functor());
218  ScalarCwiseNullaryOp xpr_scalar(lprod.rows(), lprod.cols(),
219  ScalarConstantOp(alpha));
220  ScalarTimesNewCwiseBinaryOp xpr(xpr_scalar, xpr_prod, ScalarTimesOp());
221  Assignment<Dest, ScalarTimesNewCwiseBinaryOp, addAssignOp,
222  Dense2Dense>::run(dst, xpr, addAssignOp());
