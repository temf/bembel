24 inline std::vector<Patch> LoadGeometryFile(
25  const std::string &file_name) noexcept {
26  std::vector<Bembel::Patch> out;
27  std::stringstream iss;
28  std::string word;
29  std::string row;
30  int infoInt[5];
31  std::ifstream file;
32  file.open(file_name);
33  if (!file) {
34  std::cerr << "File " << file_name << " doesn't exist!";
35  exit(1);
36  }
37  // first 4 rows are irrelevant
38  for (int i = 0; i < 5; i++) {
39  getline(file, row);
40  }
41  // main information
42  iss.str(row);
43  for (int i = 0; i < 5; i++) {
44  iss >> word;
45 
46  infoInt[i] = stoi(word);
47  }
48  // main loop - patches
49  for (int patchNr = 1; patchNr <= infoInt[2]; patchNr++) {
50  Bembel::Patch tempPatch;
51  std::vector<double> tempknt1;
52  std::vector<double> tempknt2;
53  std::vector<int> info; // p and ncp / 0,1-p 2,3-ncp
54  std::vector<Eigen::MatrixXd> tmp;
55 
56  getline(file, row); // file_name
57 
58  getline(file, row);
59  iss.str(row);
60  while (iss >> word) {
61  info.push_back(stoi(word));
62  }
63  getline(file, row);
64 
65  word = "";
66  iss.clear();
67  iss.str(row);
68  while (iss >> word) {
69  info.push_back(stoi(word));
70  }
71 
72  word = "";
73  iss.clear();
74 
75  // In textfiles are only 2 knotVectors and 4 Matrices
76 
77  getline(file, row);
78  iss.str(row);
79 
80  // first knotVector
81 
82  for (int k = 0; k < info[0] + info[2] + 1; k++) {
83  iss >> word;
84  tempknt1.push_back(atof(word.c_str()));
85  }
86  iss.clear();
87  getline(file, row);
88  iss.str(row);
89  word = "";
90  // second knotVector
91  for (int k = 0; k < info[1] + info[3] + 1; k++) {
92  iss >> word;
93  tempknt2.push_back(atof(word.c_str()));
94  }
95  iss.clear();
96  // 4 Matrices
97  int N = info[2];
98  int M = info[3];
99  for (int k = 0; k < 4; k++) {
100  Eigen::Matrix<double, -1, -1> tempMatrix(
101  M, N); // == Eigen::MatrixXd tempMatrix(M,N);
102  getline(file, row);
103  iss.str(row);
104  for (int i = 0; i < M; i++)
105  for (int j = 0; j < N; j++) {
106  iss >> word;
107 
108  tempMatrix(i, j) = atof(word.c_str());
109  }
110 
111  tmp.push_back(tempMatrix);
112  iss.clear();
113  }
114  // Important
115  tempPatch.init_Patch(tmp, tempknt1, tempknt2);
116  out.push_back(tempPatch);
117  }
118 
119  file.close();
120  return out;
121 }

153 void WritePatch(const std::string &file_name, int current_patch_number,
154  const std::vector<Eigen::MatrixXd> &xyzw,
155  const std::vector<double> &knt1,
156  const std::vector<double> &knt2) noexcept {
157  std::ofstream file(file_name, std::ios::app);
158  int N = xyzw[0].cols(); // ncp
159  int M = xyzw[0].rows(); // ncp
160  int p1 = knt1.size() - N - 1;
161  int p2 = knt2.size() - M - 1;
162  file << "PATCH " << current_patch_number << " \r\n";
163  file << p1 << " " << p2 << " \r\n";
164  file << N << " " << M << " \r\n";
165  // knotVectors
166  for (unsigned int i = 0; i < knt1.size() - 1; i++) {
167  file << std::fixed << std::setprecision(15) << knt1[i] << " ";
168  }
169  file << std::fixed << std::setprecision(15) << knt1[knt1.size() - 1]
170  << " \r\n";
171  for (unsigned int i = 0; i < knt2.size() - 1; i++) {
172  file << std::fixed << std::setprecision(15) << knt2[i] << " ";
173  }
174  file << std::fixed << std::setprecision(15) << knt2[knt2.size() - 1]
175  << " \r\n";
176 
177  // Matrices
178  for (unsigned int n = 0; n < xyzw.size(); n++) {
179  for (int i = 0; i < M; i++)
180  for (int j = 0; j < N; j++) {
181  file << std::fixed << std::setprecision(15) << xyzw[n](i, j);
182  if (N * i + j == M * N - 1) {
183  file << " \r\n";
184  continue;
185  }
186  file << " ";
187  }
188  }
189 
190  file.close();
191 }
