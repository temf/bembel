15 void computeLegendreQuadrature(Eigen::VectorXd *xi, Eigen::VectorXd *w,
16  int order) {
17  Eigen::MatrixXd A(order, order);
18  A.setZero();
19  for (auto i = 1; i < order; ++i) A(i, i - 1) = i / (sqrt(4. * i * i - 1));
20  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es;
21  es.compute(A);
22  xi->resize(order);
23  w->resize(order);
24  for (auto i = 0; i < order; ++i) {
25  (*xi)(i) = 0.5 * (es.eigenvalues()(i) + 1);
26  (*w)(i) = es.eigenvectors()(0, i) * es.eigenvectors()(0, i);
27  }
28  return;
29 }

31 Eigen::VectorXd HornersMethod(const Eigen::VectorXd &c,
32  const Eigen::VectorXd &x) {
33  Eigen::VectorXd retval(x.size());
34  retval.array() = c(0);
35  for (auto i = 1; i < c.size(); ++i) {
36  retval.array() = x.cwiseProduct(retval).array() + c(i);
37  }
38  return retval;
39 }

55  << std::endl;
56  std::cout << (w - GS[i].w_).norm() / (GS[i].w_).norm() << std::endl;
57  std::cout << "---------------\n";
58  }
59  }
60  // perform an actual quadrature test;
61  {
62  for (auto i = 0; i < max_order; ++i) {
63  Eigen::VectorXd c = Eigen::VectorXd::Random(2 * i + 2);
64  Eigen::VectorXd integral_weights =
65  1. / Eigen::ArrayXd::LinSpaced(2 * i + 2, 1, 2 * i + 2).reverse();
66  double exact_integral = c.cwiseProduct(integral_weights).sum();
67  std::cout << "degree of test polynomial: " << 2 * i + 1
68  << " integral: " << exact_integral << std::endl;
69  for (auto j = 0; j < max_order; ++j) {
70  Eigen::VectorXd pol_eval = HornersMethod(c, GS[j].xi_);
71  double quadrature_val = (pol_eval.array() * GS[j].w_.array()).sum();
72  std::cout << "quadrature degree (i.e. #qpoints) " << j + 1 << " error: "
73  << std::abs(quadrature_val - exact_integral) /
74  std::abs(exact_integral)
75  << std::endl;
76  }
77  std::cout << "---------------\n";
78  }
