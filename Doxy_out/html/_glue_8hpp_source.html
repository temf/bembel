90  Eigen::SparseMatrix<double> assembleGlueMatrix(
91  const SuperSpace<Derived> &superspace, const Projector<Derived> &proj) {
92  const int pre_dofs = proj.get_dofs_after_projector();
93  // The dofs that need to be identified with each other are divided into
94  // master and slaves, where the master is the firs w.r.t. the tp-ordering
95  std::vector<Eigen::Triplet<double>> trips;
96  std::vector<GlueRoutines::dofIdentification> dof_id =
97  makeDofIdentificationList(superspace, proj);
98 
99  // We keep track of certain information
100  std::vector<bool> dof_is_slave(pre_dofs, false);
101  std::vector<bool> dof_is_master(pre_dofs, false);
102 
103  // Here, we fill the two vectors above
104  int number_of_slaves = 0;
105  for (auto dofset : dof_id) {
106  // This sorting is required, since by construction only dofs[0]<dofs[1] is
107  // given, but in the case of corners and H1 discretizations, it might
108  // happen that dofs[0]>dofs[2]. Moreover, we count how many dofs we "glue
109  // away".
110  std::sort(dofset.dofs.begin(), dofset.dofs.end());
111  dof_is_master[dofset.dofs[0]] = true;
112  for (int i = 1; i < dofset.dofs.size(); ++i) {
113  dof_is_slave[dofset.dofs[i]] = true;
114  ++number_of_slaves;
115  }
116  }
117 
118  // Now, we sort w.r.t. the masters.
119  std::sort(dof_id.begin(), dof_id.end(),
120  [](GlueRoutines::dofIdentification a,
121  GlueRoutines::dofIdentification b) {
122  return a.dofs[0] < b.dofs[0];
123  });
124 
125  const int post_dofs = pre_dofs - number_of_slaves;
126 
127  // This block is the heart of the algorithms. Skip keeps track of how many
128  // slaves have been skipped already, and master_index keeps track on which
129  // master is about to be glued next.
130  int skip = 0;
131  int master_index = 0;
132  for (int i = 0; i < post_dofs; ++i) {
133  const int post_index = i;
134  while (dof_is_slave[i + skip] && ((i + skip) < pre_dofs)) ++skip;
135  const int pre_index = i + skip;
136  trips.push_back(Eigen::Triplet<double>(pre_index, post_index, 1));
137  // The dof cannot be declared slave and master at the same time
138  assert(!(dof_is_master[pre_index] && dof_is_slave[pre_index]));
139  if (dof_is_master[pre_index]) {
140  // The dofs in dof_id don't know they might be moved forward. So the
141  // smallest dof of those to be identified should coincide with the
142  // pre_index.
143  assert(pre_index == dof_id[master_index].dofs[0]);
144  const int number_of_partners = dof_id[master_index].dofs.size();
145  for (int j = 1; j < number_of_partners; ++j) {
146  trips.push_back(Eigen::Triplet<double>(dof_id[master_index].dofs[j],
147  post_index,
148  dof_id[master_index].coef));
149  }
150  ++master_index;
151  }
152  }
153 
154  Eigen::SparseMatrix<double> glue_matrix(pre_dofs, post_dofs);
155  glue_matrix.setFromTriplets(trips.begin(), trips.end());
156 
157  assert(trips.size() == pre_dofs);
158  return glue_matrix;
159  }

164 // This routine collects the indices of DOFs with support on an edge in
165 // dependence of the dimension of the space in each tensor product direction
166 // and the edge. The shift parameter can be used to shift it to the correct
167 // patch or vector component.
168 std::vector<int> getEdgeDofIndices(int edgeCase, int dimXdir, int dimYdir,
169  int shift) {
170  std::vector<int> out;
171  switch (edgeCase) {
172  case (0): {
173  out.reserve(dimXdir);
174  assert(
175  "This should be a given. If not something went wrong in Glue.hpp" &&
176  dimXdir <= dimYdir);
177  for (int i = 0; i < dimXdir; ++i) {
178  out.push_back(i + shift);
179  }
180  return out;
181  };
182  case (1): {
183  out.reserve(dimYdir);
184  assert(
185  "This should be a given. If not something went wrong in Glue.hpp" &&
186  dimYdir <= dimXdir);
187  for (int i = 0; i < dimYdir; ++i) {
188  out.push_back(dimXdir * (i + 1) - 1 + shift);
189  }
190  return out;
191  };
192  case (2): {
193  out.reserve(dimXdir);
194  assert(
195  "This should be a given. If not something went wrong in Glue.hpp" &&
196  dimXdir <= dimYdir);
197  for (int i = 0; i < dimXdir; ++i) {
198  out.push_back(dimXdir * (dimYdir - 1) + i + shift);
199  }
200  return out;
201  };
202  case (3): {
203  out.reserve(dimYdir);
204  assert(
205  "This should be a given. If not something went wrong in Glue.hpp" &&
206  dimYdir <= dimXdir);
207  for (int i = 0; i < dimYdir; ++i) {
208  out.push_back(dimXdir * i + shift);
209  }
210  return out;
211  };
212  default: {
213  };
214  // An edge might have a -1 index. This occurs only when no partner could
215  // be found, and the -1 is the placeholder of the missing partner.
216  }
217  return out;
218 }

220 // The following 7 functions figure out how edges and the vector components
221 // are oriented w.r.t. each other.
222 inline bool edgeIsForwardParametrized(int edgeCase) {
223  return edgeCase == 0 || edgeCase == 1;
224 }
225 inline bool edgeIsBackwardsParametrized(int edgeCase) {
226  return !(edgeIsForwardParametrized(edgeCase));
227 }
228 inline bool normalComponentIsInwardDirected(int edgeCase) {
229  return edgeCase == 0 || edgeCase == 3;
230 }
231 inline bool normalComponentIsOutwardDirected(int edgeCase) {
232  return !(normalComponentIsInwardDirected(edgeCase));
233 }
234 
235 inline bool reverseParametrized(const std::array<int, 4> &edge) {
236  return ((edgeIsForwardParametrized(edge[2]) &&
237  edgeIsForwardParametrized(edge[3])) ||
238  (edgeIsBackwardsParametrized(edge[2]) &&
239  edgeIsBackwardsParametrized(edge[3])))
240  ? true
241  : false;
242 }
243 
244 inline int glueCoefficientDivergenceConforming(const std::array<int, 4> &edge) {
245  return ((normalComponentIsInwardDirected(edge[2]) &&
246  normalComponentIsInwardDirected(edge[3])) ||
247  (normalComponentIsOutwardDirected(edge[2]) &&
248  normalComponentIsOutwardDirected(edge[3])))
249  ? -1
250  : 1;
251 }
252 inline bool edgeToBeGluedInFirstComp(int edgeCase) {
253  return (edgeCase == 0 || edgeCase == 2) ? false : true;
254 }

270  The scalar case: Here, we have only one vector component to worry about.
271  However, we need to take care of edges, since here multiple (more than 2) dofs

277 template <typename Derived>
278 struct glue_identificationmaker_<Derived, DifferentialForm::Continuous> {
279  static std::vector<dofIdentification> makeIdentification(
280  const std::vector<std::array<int, 4>> &edges_,
281  const SuperSpace<Derived> &superspace, const Projector<Derived> &proj) {
282  // We check if the space can even be continuous globally.
283  assert(superspace.get_polynomial_degree() >= 1);
284  const int one_d_dim = superspace.get_polynomial_degree() + 1 +
285  proj.get_knot_repetition() *
286  ((1 << superspace.get_refinement_level()) - 1);
287  const int dofs_per_patch = one_d_dim * one_d_dim;
288 
289  std::vector<GlueRoutines::dofIdentification> out;
290  out.reserve(edges_.size() * one_d_dim);
291 
292  std::vector<int> already_stored_in(proj.get_dofs_after_projector(), -1);
293  int d_count = 0;
294  for (auto edge : edges_) {
295  // The shift is to account for different patches, since the
296  // getEdgeDof-routines only can enumerate w.r.t. patch 0.
297  const int shift_e1 = edge[0] * dofs_per_patch;
298  const int shift_e2 = edge[1] * dofs_per_patch;
299  const bool needReversion = reverseParametrized(edge);
300 
301  std::vector<int> dofs_e1 =
302  getEdgeDofIndices(edge[2], one_d_dim, one_d_dim, shift_e1);
303  std::vector<int> dofs_e2 =
304  getEdgeDofIndices(edge[3], one_d_dim, one_d_dim, shift_e2);
305 
306  // Check if the edge is hanging. For screens one would need an "else".
307  if (edge[1] > -1 && edge[0] > -1) {
308  for (int i = 0; i < one_d_dim; ++i) {
309  GlueRoutines::dofIdentification d;
310  const int j = needReversion ? one_d_dim - 1 - i : i;
311  // const int j = i;
312  assert(dofs_e1[i] != dofs_e2[j] &&
313  "If this happens something went horribly wrong.");
314 
315  const int small_dof = std::min(dofs_e1[i], dofs_e2[j]);
316  const int large_dof = std::max(dofs_e1[i], dofs_e2[j]);
317 
318  // In the continuous case, more than two dofs might be glued together.
319  // Therefore, we must check if we know one of the dofs already. If
320  // yes, we just add the new one to the dof_id, if not, we make a new
321  // identification group.
322  if (already_stored_in[small_dof] > -1) {
323  // It could be that the large_dof is already matched with another
324  // dof, i.e., in the case of a 'circle'. Then there is nothing to
325  // do. If not, we match it with the master of the partner.
326  if (already_stored_in[large_dof] == -1) {
327  out[already_stored_in[small_dof]].dofs.push_back(large_dof);
328  already_stored_in[large_dof] = already_stored_in[small_dof];
329  } else {
330  if (!(already_stored_in[small_dof] ==
331  already_stored_in[large_dof])) {
332  // This case is tricky. Assume that we have to identify four
333  // DOFs with each other, but they have already been assigned in
334  // pairs of two. Then we need to reverse this process. First, we
335  // grab the two storage locations.
336  const int small_store = std::min(already_stored_in[large_dof],
337  already_stored_in[small_dof]);
338  const int large_store = std::max(already_stored_in[large_dof],
339  already_stored_in[small_dof]);
340  for (auto dfs : out[large_store].dofs) {
341  // now we put all of those in the larger location into the
342  // smaller one.
343  already_stored_in[dfs] = small_store;
344  for (auto other_dfs : out[small_store].dofs) {
345  assert(dfs != other_dfs);
346  }
347  out[small_store].dofs.push_back(dfs);
348  }
349  // Now we set the larger storage location to empty.
350  out[large_store].dofs = {};
351  out[large_store].coef = 0;
352  }
353  }
354  } else if (already_stored_in[large_dof] > -1) {
355  // It could be that the small_dof is already matched with another
356  // dof, i.e., in the case of a 'circle'. Then there is nothing to
357  // do. If not, we match it with the master of the partner.
358  if (already_stored_in[small_dof] == -1) {
359  out[already_stored_in[large_dof]].dofs.push_back(small_dof);
360  already_stored_in[small_dof] = already_stored_in[large_dof];
361  } else {
362  if (!(already_stored_in[small_dof] ==
363  already_stored_in[large_dof])) {
364  // This case is tricky. Assume that we have to identify four
365  // DOFs with each other, but they have already been assigned in
366  // pairs of two. Then we need to reverse this process. First, we
367  // grab the two storage locations.
368  const int small_store = std::min(already_stored_in[large_dof],
369  already_stored_in[small_dof]);
370  const int large_store = std::max(already_stored_in[large_dof],
371  already_stored_in[small_dof]);
372  for (auto dfs : out[large_store].dofs) {
373  // now we put all of those in the larger location into the
374  // smaller one.
375  already_stored_in[dfs] = small_store;
376  for (auto other_dfs : out[small_store].dofs) {
377  assert(dfs != other_dfs);
378  }
379  out[small_store].dofs.push_back(dfs);
380  }
381  // Now we set the larger storage location to empty.
382  out[large_store].dofs = {};
383  out[large_store].coef = 0;
384  }
385  }
386  } else {
387  // With the exception of corners, this will be the default case.
388  // We just add a pair of dofs to the bookkeeping.
389  d.dofs.push_back(small_dof);
390  d.dofs.push_back(large_dof);
391  already_stored_in[small_dof] = d_count;
392  already_stored_in[large_dof] = d_count;
393  d_count++;
394  d.coef = 1;
395  out.push_back(d);
396  }
397  }
398  }
399  }
400  // Now we need to clean up the empty dofsets, since subsequent routines
401  // assume there to be at least one element.
402  for (auto x = out.begin(); x != out.end(); ++x) {
403  if ((*x).dofs.size() == 0) {
404  out.erase(x);
405  }
406  }
407  return out;
408  }
409 };

414  == py == p.

417 template <typename Derived>
418 struct glue_identificationmaker_<Derived, DifferentialForm::DivConforming> {
419  static std::vector<dofIdentification> makeIdentification(
420  const std::vector<std::array<int, 4>> &edges_,
421  const SuperSpace<Derived> &superspace, const Projector<Derived> &proj) {
422  // since we assume px == py and uniform refinement in both directions, there
423  // will be a small_dim and a large_dim in every vector component.
424  const int small_dim = superspace.get_polynomial_degree() +
425  proj.get_knot_repetition() *
426  ((1 << superspace.get_refinement_level()) - 1);
427  const int large_dim = superspace.get_polynomial_degree() + 1 +
428  proj.get_knot_repetition() *
429  ((1 << superspace.get_refinement_level()) - 1);
430  const int dofs_per_patch_per_component = small_dim * large_dim;
431  // sanity check
432  assert(dofs_per_patch_per_component ==
433  (proj.get_dofs_after_projector() /
434  (superspace.get_number_of_patches() * 2)) &&
435  "The assembly of the glue matrix is highly specific to the space. "
436  "Something went wrong; is the discrete space correct?");
437 
438  std::vector<GlueRoutines::dofIdentification> out;
439  out.reserve(edges_.size() * small_dim);
440 
441  for (auto edge : edges_) {
442  assert(edge[0] <= edge[1] || edge[1] == -1);
443  // The DOFs that need to be glued on edges 0 and 2 are in the second
444  // vector component. The remainder of the dof-index-shift is to account
445  // for dofs on other patches.
446  const int shift_e1 = edge[0] * dofs_per_patch_per_component +
447  (edgeToBeGluedInFirstComp(edge[2])
448  ? 0
449  : (dofs_per_patch_per_component *
450  superspace.get_number_of_patches()));
451  const int shift_e2 = edge[1] * dofs_per_patch_per_component +
452  (edgeToBeGluedInFirstComp(edge[3])
453  ? 0
454  : (dofs_per_patch_per_component *
455  superspace.get_number_of_patches()));
456 
457  const int xdir_dim_1 =
458  edgeToBeGluedInFirstComp(edge[2]) ? large_dim : small_dim;
459  const int ydir_dim_1 =
460  edgeToBeGluedInFirstComp(edge[2]) ? small_dim : large_dim;
461  const int xdir_dim_2 =
462  edgeToBeGluedInFirstComp(edge[3]) ? large_dim : small_dim;
463  const int ydir_dim_2 =
464  edgeToBeGluedInFirstComp(edge[3]) ? small_dim : large_dim;
465  std::vector<int> dofs_e1 =
466  getEdgeDofIndices(edge[2], xdir_dim_1, ydir_dim_1, shift_e1);
467  std::vector<int> dofs_e2 =
468  getEdgeDofIndices(edge[3], xdir_dim_2, ydir_dim_2, shift_e2);
469 
470  const int size_of_edge_dofs = dofs_e1.size();
471  assert(size_of_edge_dofs == dofs_e2.size() || edge[1] == -1);
472 
473  const bool needReversion = reverseParametrized(edge);
474  const int coef = glueCoefficientDivergenceConforming(edge);
475 
476  // Check if the edge is hanging. For screens one would need an "else".
477  if (edge[1] > -1 && edge[0] > -1) {
478  for (int i = 0; i < size_of_edge_dofs; ++i) {
479  GlueRoutines::dofIdentification d;
480  const int j = needReversion ? size_of_edge_dofs - 1 - i : i;
481  assert(dofs_e1[i] != dofs_e2[j] &&
482  "If this happens something went horribly wrong.");
483  d.dofs.push_back(std::min(dofs_e1[i], dofs_e2[j]));
484  d.dofs.push_back(std::max(dofs_e1[i], dofs_e2[j]));
485  d.coef = coef;
486  out.push_back(d);
487  }
488  }
489  }
490  out.shrink_to_fit();
491  return out;
492  }
