84  Eigen::Vector2d a) {

87  PSpecificShapeFunctionHandler<I - 1>::evalBasis(I - 1, X, a(0));

138 template <int P, typename Scalar>
139 void Phi_times_Phi_(Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> *c,
140  Scalar w, Eigen::Vector2d xi, Eigen::Vector2d eta) {
141  constexpr int I = P + 1;
142  Scalar a[I * I];
143  double b[I * I], X[I], Y[I];
144 
145  PSpecificShapeFunctionHandler<I - 1>::evalBasis(I - 1, X, xi(0));
146  PSpecificShapeFunctionHandler<I - 1>::evalBasis(I - 1, Y, xi(1));
147 
148  for (int iy = 0; iy < I; iy++)
149  for (int ix = 0; ix < I; ix++) a[iy * I + ix] = w * X[ix] * Y[iy];
150 
151  PSpecificShapeFunctionHandler<I - 1>::evalBasis(I - 1, X, eta(0));
152  PSpecificShapeFunctionHandler<I - 1>::evalBasis(I - 1, Y, eta(1));
153 
154  for (int iy = 0; iy < I; iy++)
155  for (int ix = 0; ix < I; ix++) b[iy * I + ix] = X[ix] * Y[iy];
156 
157  for (int i = 0; i < (I * I); i++)
158  for (int j = 0; j < (I * I); j++) (*c)(i, j) += a[i] * b[j];
159 
160  return;
161 }

167 template <int P, typename Scalar>
168 void Div_Phi_times_Div_Phi_(
169  Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> *c, Scalar weight,
170  Eigen::Vector2d xi, Eigen::Vector2d eta) {
171  constexpr int I = P + 1;
172  constexpr int I2 = I * I;
173  Eigen::Matrix<Scalar, Eigen::Dynamic, 1> a_dx(I2);
174  a_dx.setZero();
175  Eigen::Matrix<Scalar, Eigen::Dynamic, 1> a_dy(I2);
176  a_dy.setZero();
177  Eigen::VectorXd b_dx(I2);
178  b_dx.setZero();
179  Eigen::VectorXd b_dy(I2);
180  b_dy.setZero();
181 
182  phiphi_dx_<P>(&a_dx, weight, xi);
183  phiphi_dy_<P>(&a_dy, weight, xi);
184  phiphi_dx_<P>(&b_dx, 1., eta);
185  phiphi_dy_<P>(&b_dy, 1., eta);
186 
187  for (int i = 0; i < I2; ++i)
188  for (int j = 0; j < I2; ++j) (*c)(i, j) += a_dx[i] * b_dx[j];
189  for (int i = 0; i < I2; ++i)
190  for (int j = 0; j < I2; ++j) (*c)(i, j + I2) += a_dx[i] * b_dy[j];
191  for (int i = 0; i < I2; ++i)
192  for (int j = 0; j < I2; ++j) (*c)(i + I2, j) += a_dy[i] * b_dx[j];
193  for (int i = 0; i < I2; ++i)
194  for (int j = 0; j < I2; ++j) (*c)(i + I2, j + I2) += a_dy[i] * b_dy[j];
195 
196  return;
197 }
