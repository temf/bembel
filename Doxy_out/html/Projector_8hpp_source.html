98 template <typename Derived>
99 inline _proj_info makeLocalProjectionTriplets(
100  const SuperSpace<Derived>& super_space, const int pp1x, const int pp1y,
101  const int knotrepetition_in) {
102  const int M = super_space.get_refinement_level();
103  const int maximal_polynomial_degree = std::max(pp1x, pp1y);
104  assert(maximal_polynomial_degree == super_space.get_polynomial_degree() + 1 &&
105  "superSpace not suitable for desired ansatz space -> polynomial "
106  "degree mismatch");
107  const int minp = std::min(pp1x, pp1y);
108  const int knotrepetition = (maximal_polynomial_degree <= knotrepetition_in)
109  ? minp
110  : knotrepetition_in;
111  const int n = (1 << M);
112  const int patch_number = super_space.get_number_of_patches();
113 
114  // At the moment, we allow a difference of one only, i.e., we start our de
115  // Rham sequence with a space of the same degree in every TP-direction.
116  assert(std::abs(maximal_polynomial_degree - minp) <= 1);
117 
118  // Now, we construct the continuous spaces which will be the preimage of the
119  // projector. n-1 is passed, since n = number_elements but n-1 = number of
120  // interior knots.
121  std::vector<double> c_space_knot_x =
122  Spl::MakeUniformKnotVector(pp1x, n - 1, knotrepetition);
123  std::vector<double> c_space_knot_y =
124  Spl::MakeUniformKnotVector(pp1y, n - 1, knotrepetition);
125  const int c_space_dim_x = c_space_knot_x.size() - pp1x;
126  const int c_space_dim_y = c_space_knot_y.size() - pp1y;
127  const int c_space_dim = c_space_dim_x * c_space_dim_y;
128 
129  _proj_info out;
130 
131  out.cols.reserve(c_space_dim * maximal_polynomial_degree *
132  maximal_polynomial_degree * patch_number);
133  out.vals.reserve(c_space_dim * maximal_polynomial_degree *
134  maximal_polynomial_degree * patch_number);
135  out.rows.reserve(c_space_dim * maximal_polynomial_degree *
136  maximal_polynomial_degree * patch_number);
137 
138  std::vector<double> mask =
139  Spl::MakeInterpolationMask(maximal_polynomial_degree);
140  const int masksize = mask.size();
141 
142  // Now we assemble our system which we use for the interpolation
143  assert(masksize == maximal_polynomial_degree &&
144  "projector.cpp: System needs to be square");
145  Eigen::Matrix<double, -1, -1> system(
146  masksize * masksize,
147  maximal_polynomial_degree * maximal_polynomial_degree);
148 
149  // Here, we suddenly use the degree for basisevaluation, i.e.,
150  // maximal_polynomial_degree-1. This is confusing, but correct and tested.
151  {
152  double vals_y[Constants::MaxP + 1];
153  double vals_x[Constants::MaxP + 1];
154  for (int iy = 0; iy < maximal_polynomial_degree; ++iy) {
155  Bembel::Basis::ShapeFunctionHandler::evalBasis(
156  maximal_polynomial_degree - 1, vals_y, mask[iy]);
157  for (int ix = 0; ix < maximal_polynomial_degree; ++ix) {
158  Bembel::Basis::ShapeFunctionHandler::evalBasis(
159  maximal_polynomial_degree - 1, vals_x, mask[ix]);
160  for (int jy = 0; jy < maximal_polynomial_degree; ++jy) {
161  for (int jx = 0; jx < maximal_polynomial_degree; ++jx) {
162  system(iy * masksize + ix, jy * maximal_polynomial_degree + jx) =
163  vals_x[jx] * vals_y[jy];
164  }
165  }
166  }
167  }
168  }
169 
170  // Do LU-Decomposition of
171  // systemsuper_space.get_mesh().get_element_tree().begin()
172  const Eigen::PartialPivLU<Eigen::Matrix<double, -1, -1>> pplu(system);
173 
174  unsigned int element_number = 0;
175 
176  for (auto element = super_space.get_mesh().get_element_tree().cpbegin();
177  element != super_space.get_mesh().get_element_tree().cpend();
178  ++element) {
179  // s = x
180  const double pos_x = element->llc_(0);
181  const double pos_y = element->llc_(1);
182 
183  const double h = element->get_h();
184  const double mid_x = pos_x + (.5 * h);
185  const double mid_y = pos_y + (.5 * h);
186 
187  // The following block of code computes the indeces of all continuous basis
188  // functions which have a support on the element of index 'element'
189  std::vector<double> nonzero_dofs_x;
190  std::vector<double> nonzero_dofs_y;
191  nonzero_dofs_x.reserve(pp1x);
192  nonzero_dofs_y.reserve(pp1y);
193  for (int dof_y = 0; dof_y < c_space_dim_y; ++dof_y) {
194  if ((c_space_knot_y[dof_y] <= mid_y) &&
195  (c_space_knot_y[dof_y + pp1y] >= mid_y)) {
196  nonzero_dofs_y.push_back(dof_y);
197  }
198  }
199  for (int dof_x = 0; dof_x < c_space_dim_x; ++dof_x) {
200  if ((c_space_knot_x[dof_x] <= mid_x) &&
201  (c_space_knot_x[dof_x + pp1x] >= mid_x)) {
202  nonzero_dofs_x.push_back(dof_x);
203  }
204  }
205  nonzero_dofs_x.shrink_to_fit();
206  nonzero_dofs_y.shrink_to_fit();
207 
208  // We now loop over all basis functions wich are non-zero on the element
209  for (auto dof_y : nonzero_dofs_y) {
210  for (auto dof_x : nonzero_dofs_x) {
211  // The next few lines are to check if the element is within the support
212  // of the basis function given by dof_x and dof_y.
213  std::vector<double> local_mask_x(masksize);
214  std::vector<double> local_mask_y(masksize);
215  for (int i = 0; i < masksize; ++i) {
216  local_mask_x[i] = pos_x + (1.0 / n) * mask[i];
217  local_mask_y[i] = pos_y + (1.0 / n) * mask[i];
218  assert(local_mask_x[i] < 1 && local_mask_x[i] > 0);
219  assert(local_mask_y[i] < 1 && local_mask_y[i] > 0);
220  }
221 
222  // build unit vectors
223  std::vector<double> c_coefs_x(c_space_dim_x, 0.0);
224  std::vector<double> c_coefs_y(c_space_dim_y, 0.0);
225  c_coefs_x[dof_x] = 1.;
226  c_coefs_y[dof_y] = 1.;
227 
228  // evaluate rhs for interpolation problem
229  std::vector<double> vals_x =
230  Spl::DeBoor(c_coefs_x, c_space_knot_x, local_mask_x);
231  std::vector<double> vals_y =
232  Spl::DeBoor(c_coefs_y, c_space_knot_y, local_mask_y);
233 
234  // assemble and solve interpolation problem
235  Eigen::VectorXd rhs(masksize * masksize);
236  for (int iy = 0; iy < masksize; ++iy)
237  for (int ix = 0; ix < masksize; ++ix)
238  rhs(iy * masksize + ix) = vals_x[ix] * vals_y[iy];
239  Eigen::VectorXd sol = pplu.solve(rhs);
240 
241  // insert entries into helperstruct
242  for (int i = 0;
243  i < maximal_polynomial_degree * maximal_polynomial_degree; ++i) {
244  if (std::abs(sol(i)) > Constants::projector_tolerance) {
245  out.cols.push_back(element->patch_ * (c_space_dim) +
246  (c_space_dim_x * dof_y) + dof_x);
247  out.rows.push_back((element_number * maximal_polynomial_degree *
248  maximal_polynomial_degree) +
249  i);
250  out.vals.push_back(sol(i));
251  }
252  }
253  }
254  }
255  ++element_number;
256  }
257 
258  out.num_c_dof = c_space_dim_x * c_space_dim_y * patch_number;
259  out.num_dc_dof = maximal_polynomial_degree * maximal_polynomial_degree * n *
260  n * patch_number;
261 
262  out.cols.shrink_to_fit();
263  out.rows.shrink_to_fit();
264  out.vals.shrink_to_fit();
265 
266  assert((out.vals.size() == out.rows.size()) &&
267  (out.rows.size() == out.cols.size()) &&
268  "projector.cpp: you made a mistake, try again.");
269 
270  return out;
271 }

275  static Eigen::SparseMatrix<double> makeMatrix(
276  const SuperSpace<Derived>& super_space, const int knotrepetition) {
277  const int P = super_space.get_polynomial_degree();
278  assert(P > 0 && "P must be 1 or larger for this type of discrete space");
279  assert(knotrepetition <= P &&
280  "Knot repetition must be smaller than P for this type of discrete "
281  "space");

283  Eigen::SparseMatrix<double> local_matrix = projector_matrixmaker_<
284  Derived, DifferentialForm::Discontinuous>::makeMatrix(super_space,
285  knotrepetition);
286 
287  return local_matrix;
288  }

291 template <typename Derived>
292 struct projector_matrixmaker_<Derived, DifferentialForm::DivConforming> {
293  static Eigen::SparseMatrix<double> makeMatrix(
294  const SuperSpace<Derived>& super_space, const int knotrepetition) {
295  assert(super_space.get_polynomial_degree() > 0 &&
296  "P must be 1 or larger for this type of discrete space");
297  assert(knotrepetition <= super_space.get_polynomial_degree() &&
298  "Knot repetition must be smaller than P for this type of discrete "
299  "space");
300  const auto info1 = ProjectorRoutines::makeLocalProjectionTriplets<Derived>(
301  super_space, super_space.get_polynomial_degree() + 1,
302  super_space.get_polynomial_degree(), knotrepetition);
303  const auto info2 = ProjectorRoutines::makeLocalProjectionTriplets<Derived>(
304  super_space, super_space.get_polynomial_degree(),
305  super_space.get_polynomial_degree() + 1, knotrepetition);
306  const int size1 = info1.vals.size();
307  const int size2 = info2.vals.size();
308  assert(size1 == size2);
309  std::vector<Eigen::Triplet<double>> trips;
310  for (int k = 0; k < size1; ++k) {
311  trips.push_back(
312  Eigen::Triplet<double>(info1.rows[k], info1.cols[k], info1.vals[k]));
313  }
314  for (int k = 0; k < size2; ++k) {
315  trips.push_back(Eigen::Triplet<double>(info2.rows[k] + info1.num_dc_dof,
316  info2.cols[k] + info1.num_c_dof,
317  info2.vals[k]));
318  }
319  Eigen::SparseMatrix<double> local_matrix(
320  info1.num_dc_dof + info2.num_dc_dof, info1.num_c_dof + info2.num_c_dof);
321  local_matrix.setFromTriplets(trips.begin(), trips.end());
322 
323  return local_matrix;
324  }
325 };
