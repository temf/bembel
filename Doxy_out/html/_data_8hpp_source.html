20 /* @brief This function implements a harmonic function, which, in the
21  * interior dormain, satisfies the Laplace equation.
22  *
23  */
24 inline double HarmonicFunction(Eigen::Vector3d in) {
25  return (4 * in(0) * in(0) - 3 * in(1) * in(1) - in(2) * in(2));
26 }
27 /* @brief This function implements the gradient of a harmonic function,
28  * which, in the interior dormain, satisfies the Laplace equation.
29  *
30  */
31 inline Eigen::Vector3d HarmonicFunctionGrad(Eigen::Vector3d in) {
32  return Eigen::Vector3d(8 * in(0), -6 * in(1), -2 * in(2));
33 }

35 /* @brief This function implements the Helmholtz fundamental solution,
36  * which, if center is placed in the interior domain, satisfies the Helmholtz
37  * equation and radiation condition in the exterior domain.
38  *
39  * Note that the sign in the fundamental solution is linked to the sign of kappa
40  * in this function.
41  */
42 inline std::complex<double> HelmholtzFundamentalSolution(
43  Eigen::Vector3d pt, std::complex<double> kappa,
44  Eigen::Vector3d center = Eigen::Vector3d(0, 0, 0)) {
45  auto c = pt - center;
46  auto r = c.norm();
47  auto i = std::complex<double>(0., 1.);
48  return std::exp(-i * kappa * r) / r;
49 }
50 /* @brief This function implements the gradient of the Helmholtz
51  * fundamental solution, which, if center is placed in the interior domain,
52  * satisfies the Helmholtz equation and radiation condition in the exterior
53  * domain.
54  *
55  * Note that the sign in the fundamental solution is linked to the sign of kappa
56  * in this function.
57  */
58 inline Eigen::Vector3cd HelmholtzFundamentalSolutionGrad(
59  Eigen::Vector3d pt, std::complex<double> kappa,
60  Eigen::Vector3d center = Eigen::Vector3d(0, 0, 0)) {
61  auto c = pt - center;
62  auto r = c.norm();
63  auto r3 = r * r * r;
64  auto i = std::complex<double>(0., 1.);
65  return -c * std::exp(-i * kappa * r) * (1. + i * kappa * r) / r3;
66 }

68 /* @brief This function implement a Hertz Dipole as in page 411 of J.D.Jacksons
69  * "Classical Electrodynamics", 3rd ed., which, if the dipole axis given by
70  * position and length remains in the interior domain, satisfies the curl-curl
71  * equation and the radiation condition in the exterior domain.
72  *
73  * Note that the sign in the fundamental solution is linked to the sign of kappa
74  * in this function.
75  */
76 inline Eigen::Vector3cd Dipole(Eigen::Vector3d pt, std::complex<double> kappa,
77  Eigen::Vector3d position,
78  Eigen::Vector3d length) {
79  constexpr std::complex<double> i(0, 1);
80  const Eigen::Vector3d c = pt - position;
81  double r = c.norm();
82  const Eigen::Vector3d n = c / r;
83  const std::complex<double> expc = std::exp(-i * kappa * r);
84  const Eigen::Vector3cd E =
85  (kappa * kappa * expc / r) * (n.cross(length).cross(n));
86  const Eigen::Vector3d h = 3 * n.dot(length) * n - length;
87  const std::complex<double> ec =
88  ((1 / (r * r * r)) - (i * (-kappa) / (r * r))) * expc;
89  return E + ec * h;
90 }
