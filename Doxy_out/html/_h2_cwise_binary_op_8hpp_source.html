19 class CwiseBinaryOpImpl<BinaryOp, Lhs, Rhs, H2>
20  : public H2MatrixBase<CwiseBinaryOp<BinaryOp, Lhs, Rhs>> {

27  ((!internal::is_same<

30  ((internal::evaluator<Lhs>::Flags & RowMajorBit) ==

32  THE_STORAGE_ORDER_OF_BOTH_SIDES_MUST_MATCH);
33  }
34 };
