90  template <typename Derived>
91  void init_H2Matrix(const Derived& linOp,
92  const Bembel::AnsatzSpace<Derived>& ansatz_space,
93  int number_of_points = 9) {
94  // get transformation matrix from ansatz space
95  transformation_matrix_ = ansatz_space.get_transformation_matrix();
101  // initialize block-cluster-trees and get parameters
102  const int vector_dimension = Bembel::getFunctionSpaceVectorDimension<
103  Bembel::LinearOperatorTraits<Derived>::Form>();
104  block_cluster_tree_.resize(vector_dimension, vector_dimension);
105  {
106  Bembel::BlockClusterTree<Scalar> bt(linOp, ansatz_space);
107  for (int i = 0; i < vector_dimension; ++i)
108  for (int j = 0; j < vector_dimension; ++j)
109  block_cluster_tree_(i, j) =
110  bt; // .init_BlockClusterTree(linOp, ansatz_space);
111  }
112  auto parameters = block_cluster_tree_(0, 0).get_parameters();
113  // compute transfer and moment matrices for fmm
114  int cluster_level = parameters.max_level_ - parameters.min_cluster_level_;
115  if (cluster_level < 0) cluster_level = 0;
116  int cluster_refinement = parameters.min_cluster_level_;
117  if (cluster_refinement > parameters.max_level_)
118  cluster_refinement = parameters.max_level_;
119  fmm_transfer_matrices_ = Bembel::H2Multipole::computeTransferMatrices<
120  Bembel::H2Multipole::ChebychevRoots>(number_of_points);
121  fmm_moment_matrix_ = Bembel::H2Multipole::
122  Moment2D<Bembel::H2Multipole::ChebychevRoots, Derived>::compute2DMoment(
123  ansatz_space.get_superspace(), cluster_level, cluster_refinement,
124  number_of_points);
125  // compute interpolation points
126  Eigen::VectorXd interpolation_points1D =
127  Bembel::H2Multipole::ChebychevRoots(number_of_points).points_;
128  Eigen::MatrixXd interpolation_points2D =
129  Bembel::H2Multipole::interpolationPoints2D(interpolation_points1D);
130  // compute content of tree leafs
131  int polynomial_degree = ansatz_space.get_polynomial_degree();
132  int polynomial_degree_plus_one_squared =
133  (polynomial_degree + 1) * (polynomial_degree + 1);
134  Bembel::GaussSquare<Bembel::Constants::maximum_quadrature_degree> GS;
135  auto super_space = ansatz_space.get_superspace();
136  auto ffield_deg = linOp.get_FarfieldQuadratureDegree(polynomial_degree);
137  std::vector<ElementSurfacePoints> ffield_qnodes =
138  Bembel::DuffyTrick::computeFfieldQnodes(super_space, GS[ffield_deg]);
139  const int NumberOfFMMComponents =
140  Bembel::LinearOperatorTraits<Derived>::NumberOfFMMComponents;
141 #pragma omp parallel
142  {
143 #pragma omp single
144  {
145  // build vector of iterators
146  std::vector<
147  typename std::vector<Bembel::BlockClusterTree<Scalar>*>::iterator>
148  leafs;
149  leafs.resize(vector_dimension * vector_dimension);
150  for (int i = 0; i < vector_dimension; ++i)
151  for (int j = 0; j < vector_dimension; ++j)
152  leafs[i * vector_dimension + j] =
153  block_cluster_tree_(j, i).lbegin();
154  // iterate over leafs
155  for (; leafs[0] != block_cluster_tree_(0, 0).lend();) {
156 #pragma omp task firstprivate(leafs)
157  {
158  switch ((*(leafs[0]))->get_cc()) {
159  // assemble dense matrix blocks
160  case Bembel::BlockClusterAdmissibility::Dense: {
161  const Bembel::ElementTreeNode* cluster1 =
162  (*(leafs[0]))->get_cluster1();
163  const Bembel::ElementTreeNode* cluster2 =
164  (*(leafs[0]))->get_cluster2();
165  int block_size =
166  std::distance(cluster2->begin(), cluster2->end()) *
167  polynomial_degree_plus_one_squared;
168  std::vector<
169  Eigen::Matrix<ScalarT, Eigen::Dynamic, Eigen::Dynamic>>
170  F;
171  for (int i = 0; i < vector_dimension * vector_dimension; ++i)
172  F.push_back(
173  Eigen::Matrix<ScalarT, Eigen::Dynamic, Eigen::Dynamic>(
174  block_size, block_size));
175  // iterate over elements in dense matrix block
176  unsigned int cl1index = 0;
177  unsigned int cl2index = 0;
178  for (const auto& element1 : *cluster1) {
179  cl2index = 0;
180  for (const auto& element2 : *cluster2) {
181  Eigen::Matrix<ScalarT, Eigen::Dynamic, Eigen::Dynamic>
182  intval(vector_dimension *
183  polynomial_degree_plus_one_squared,
184  vector_dimension *
185  polynomial_degree_plus_one_squared);
186  // do integration
187  Bembel::DuffyTrick::evaluateBilinearForm(
188  linOp, super_space, element1, element2, GS,
189  ffield_qnodes[element1.id_],
190  ffield_qnodes[element2.id_], &intval);
191  // insert into dense matrices of all block cluster trees
192  for (int i = 0; i < vector_dimension; ++i)
193  for (int j = 0; j < vector_dimension; ++j)
194  F[i * vector_dimension + j].block(
195  polynomial_degree_plus_one_squared * cl1index,
196  polynomial_degree_plus_one_squared * cl2index,
197  polynomial_degree_plus_one_squared,
198  polynomial_degree_plus_one_squared) =
199  intval.block(j * polynomial_degree_plus_one_squared,
200  i * polynomial_degree_plus_one_squared,
201  polynomial_degree_plus_one_squared,
202  polynomial_degree_plus_one_squared);
203  ++cl2index;
204  }
205  ++cl1index;
206  }
207  for (int i = 0; i < vector_dimension * vector_dimension; ++i)
208  (*(leafs[i]))->get_leaf().set_F(F[i]);
209  } break;
210  // interpolation for low-rank blocks
211  case Bembel::BlockClusterAdmissibility::LowRank: {
212  auto F = Bembel::H2Multipole::interpolateKernel<Derived>(
213  linOp, super_space, interpolation_points2D,
214  *((*(leafs[0]))->get_cluster1()),
215  *((*(leafs[0]))->get_cluster2()));
216  for (int i = 0; i < vector_dimension; ++i) {
217  for (int j = 0; j < vector_dimension; ++j) {
218  (*(leafs[i * vector_dimension + j]))
219  ->get_leaf()
220  .set_F(F.block(j * interpolation_points2D.rows() *
221  NumberOfFMMComponents,
222  i * interpolation_points2D.rows() *
223  NumberOfFMMComponents,
224  interpolation_points2D.rows() *
225  NumberOfFMMComponents,
226  interpolation_points2D.rows() *
227  NumberOfFMMComponents));
228  (*(leafs[i * vector_dimension + j]))
229  ->get_leaf()
230  .set_low_rank_flag(true);
231  }
232  }
233  } break;
234  // this leaf is not a low-rank block and not a dense block,
235  // thus it is not a leaf -> error
236  default:
237  assert(0 && "This should never happen");
238  break;
239  }
240  }
241  // increment leafs of all block-cluster trees
242  for (auto leafsit = leafs.begin(); leafsit != leafs.end(); ++leafsit)
243  ++(*leafsit);
244  }
245  }
246  }
247  }

258  dense.col(i) = (*this) * unit;
259  }
260  return dense;
261  }

299 template <typename Rhs, typename ScalarT>
300 struct generic_product_impl<H2Matrix<ScalarT>, Rhs, SparseShape, DenseShape,
301  GemvProduct> // GEMV stands for matrix-vector
302  : generic_product_impl_base<H2Matrix<ScalarT>, Rhs,
303  generic_product_impl<H2Matrix<ScalarT>, Rhs>> {
304  template <typename Dest>
305  static void scaleAndAddTo(Dest& dst, const H2Matrix<ScalarT>& lhs,
306  const Rhs& rhs, const ScalarT& alpha) {
307  // This method should implement "dst += alpha * lhs * rhs" inplace, however,
308  // for iterative solvers, alpha is always equal to 1, so let's not bother
309  // about it.
310  assert(alpha == ScalarT(1) && "scaling is not implemented");
311  EIGEN_ONLY_USED_FOR_DEBUG(alpha);
312 
313  // get H2-data
314  int max_level =
315  lhs.get_block_cluster_tree()(0, 0).get_parameters().max_level_;
316  int min_cluster_level =
317  lhs.get_block_cluster_tree()(0, 0).get_parameters().min_cluster_level_;
318  auto moment_matrix = lhs.get_fmm_moment_matrix();
319  auto transfer_matrices = lhs.get_fmm_transfer_matrices();
320  int vector_dimension = moment_matrix.size();
321 
322  // go discontinuous in rhs
323  Matrix<ScalarT, Dynamic, 1> long_rhs_all =
324  (lhs.get_transformation_matrix() * rhs).eval();
325  int vector_component_size = long_rhs_all.rows() / vector_dimension;
326 
327  // initialize destination
328  Matrix<ScalarT, Dynamic, 1> long_dst_all(long_rhs_all.rows());
329  long_dst_all.setZero();
330 
331  for (int col_component = 0; col_component < vector_dimension;
332  ++col_component) {
333  for (int row_component = 0; row_component < vector_dimension;
334  ++row_component) {
335  Matrix<ScalarT, Dynamic, 1> long_rhs = long_rhs_all.segment(
336  col_component * vector_component_size, vector_component_size);
337  Matrix<ScalarT, Dynamic, 1> long_dst(long_rhs.rows());
338  long_dst.setZero();
339 
340  // split long rhs into pieces by reshaping
341  Matrix<ScalarT, Dynamic, Dynamic> long_rhs_matrix =
342  Map<Matrix<ScalarT, Dynamic, Dynamic>>(
343  long_rhs.data(), moment_matrix[col_component].cols(),
344  long_rhs.rows() / moment_matrix[col_component].cols());
345 
346  // do forward-transformation
347  std::vector<Matrix<ScalarT, Dynamic, Dynamic>> long_rhs_forward =
348  Bembel::H2Multipole::forwardTransformation(
349  moment_matrix[col_component], transfer_matrices,
350  max_level - min_cluster_level, long_rhs_matrix);
351 
352 #pragma omp parallel
353  {
354  // initialize target for each process
355  Matrix<ScalarT, Dynamic, 1> my_long_dst(long_dst.rows());
356 
357  // initialize target of backward-transformation
358  std::vector<Matrix<ScalarT, Dynamic, Dynamic>> my_long_dst_backward;
359  for (int i = 0; i < long_rhs_forward.size(); ++i)
360  my_long_dst_backward.push_back(
361  Matrix<ScalarT, Dynamic, Dynamic>::Zero(
362  long_rhs_forward[i].rows(), long_rhs_forward[i].cols()));
363  my_long_dst.setZero();
364 
365  // matrix-vector
366  for (auto leaf =
367  lhs.get_block_cluster_tree()(row_component, col_component)
368  .clbegin();
369  leaf !=
370  lhs.get_block_cluster_tree()(row_component, col_component)
371  .clend();
372  ++leaf) {
373 #pragma omp single nowait
374  {
375  switch ((*leaf)->get_cc()) {
376  // deal with matrix blocks
377  case Bembel::BlockClusterAdmissibility::Dense: {
378  my_long_dst.segment((*leaf)->get_row_start_index(),
379  (*leaf)->get_row_end_index() -
380  (*leaf)->get_row_start_index()) +=
381  (*leaf)->get_leaf().get_F() *
382  long_rhs.segment((*leaf)->get_col_start_index(),
383  (*leaf)->get_col_end_index() -
384  (*leaf)->get_col_start_index());
385  } break;
386  // deal with low-rank blocks
387  case Bembel::BlockClusterAdmissibility::LowRank: {
388  const Bembel::ElementTreeNode* cluster1 =
389  (*leaf)->get_cluster1();
390  const Bembel::ElementTreeNode* cluster2 =
391  (*leaf)->get_cluster2();
392  int cluster_level = cluster1->get_level();
393  int fmm_level = lhs.get_block_cluster_tree()(0, 0)
394  .get_parameters()
395  .max_level_ -
396  lhs.get_block_cluster_tree()(0, 0)
397  .get_parameters()
398  .min_cluster_level_ -
399  (*leaf)->get_cluster1()->get_level();
400  int cluster1_col = cluster1->id_;
401  int cluster2_col = cluster2->id_;
402  my_long_dst_backward[fmm_level].col(cluster1_col) +=
403  (*leaf)->get_leaf().get_F() *
404  long_rhs_forward[fmm_level].col(cluster2_col);
405  } break;
406  // this leaf is not a low-rank block and not a dense block, thus
407  // it is not a leaf -> error
408  default:
409  assert(0 && "This should never happen");
410  break;
411  }
412  }
413  }
414 
415  // do backward transformation
416  my_long_dst += Bembel::H2Multipole::backwardTransformation(
417  moment_matrix[row_component], transfer_matrices,
418  max_level - min_cluster_level, my_long_dst_backward);
419 
420 #pragma omp critical
421  long_dst += my_long_dst;
422  }
423 
424  // finish off vector component
425  long_dst_all.segment(row_component * vector_component_size,
426  vector_component_size) += long_dst;
427  }
428  }
429 
430  // go continuous and write output
431  dst += lhs.get_transformation_matrix().transpose() * long_dst_all;
432  }
433 };
