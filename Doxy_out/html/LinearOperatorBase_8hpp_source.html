20 template <typename Derived>
21 struct LinearOperatorBase {
22  // Constructors
23  LinearOperatorBase() {}
24  // the user has to provide the implementation of this function, which
25  // is able to evaluate the integrand of the Galerkin formulation in a
26  // pair of quadrature points represented as a
27  // Surface point [xi; h * w; Chi(xi); dsChi(xi); dtChi(xi)]
28  template <class T>
29  void evaluateIntegrand(
30  const T &super_space, const SurfacePoint &p1, const SurfacePoint &p2,
31  Eigen::Matrix<typename LinearOperatorTraits<Derived>::Scalar,
32  Eigen::Dynamic, Eigen::Dynamic> *intval) const {
33  static_cast<const Derived *>(this)->evaluateIntegrand_impl(super_space, p1,
34  p2, intval);
35  return;
36  }
37  // the user has to provide the implementation of this function, which
38  // is able to evaluate the interpolation values of the Galerkin formulation on
39  // the reference domain in a pair of interpolation points represented as a
40  // Surface point [xi; 1.; Chi(xi); dsChi(xi); dtChi(xi)]
41  Eigen::Matrix<
42  typename LinearOperatorTraits<Derived>::Scalar,
43  getFunctionSpaceVectorDimension<LinearOperatorTraits<Derived>::Form>() *
44  LinearOperatorTraits<Derived>::NumberOfFMMComponents,
45  getFunctionSpaceVectorDimension<LinearOperatorTraits<Derived>::Form>() *
46  LinearOperatorTraits<Derived>::NumberOfFMMComponents>
47  evaluateFMMInterpolation(const SurfacePoint &p1,
48  const SurfacePoint &p2) const {
49  return static_cast<const Derived *>(this)->evaluateFMMInterpolation_impl(
50  p1, p2);
51  }
52  // return the required quadrature degree for the far-field
53  int get_FarfieldQuadratureDegree(int ansatz_degree) const {
54  return ansatz_degree - LinearOperatorTraits<Derived>::OperatorOrder + 1;
55  }
65  // return the required quadrature degree for the near-field
66  int getNearfieldQuadratureDegree(int ansatz_degree, double distance,
67  int level) const {
68  // if farfield quadrature is computed: take log of distance
69  // otherwise compute quadrature degree for Duffy trick
70  double distance_log =
71  (distance * (1 << level) < 1) ? -(level * log(2.)) : log(distance);
72 
73  // alpha/2 is the convergence rate of the corresponding Galerkin solution,
74  // this ensures the correct convergence rate of the potential
75  int alpha =
76  2 - LinearOperatorTraits<Derived>::OperatorOrder + 2 * ansatz_degree;
77 
78  // compute numerator and denominator of quadrature degree
79  double numerator =
80  (alpha + ansatz_degree) * level * log(2.) -
81  (2 - ansatz_degree + LinearOperatorTraits<Derived>::OperatorOrder) *
82  distance_log;
83  double denominator = (level + 2) * log(2.) + distance_log;
84 
85  return 0.5 * numerator / denominator;
86  }
87  // pointer to the derived object
88  Derived &derived() { return *static_cast<Derived *>(this); }
89  // const pointer to the derived object
90  const Derived &derived() const { return *static_cast<const Derived *>(this); }
91 };
