143  void addScaledSurfaceCurlInteraction(
144  Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>* intval, Scalar w,
145  const SurfacePoint& p1, const SurfacePoint& p2) const {
146  // surface measures
147  double kappa1 = p1.segment<3>(6).cross(p1.segment<3>(9)).norm();
148  double kappa2 = p2.segment<3>(6).cross(p2.segment<3>(9)).norm();
149  // compute basis functions's surface curl. Each column of s_curl is a basis
150  // function's surface curl at point s.
151  Eigen::MatrixXd s_curl(3, polynomial_degree_plus_one_squared);
152  s_curl = (1.0 / kappa1) *
153  (-p1.segment<3>(6) * basisDy(p1.segment<2>(0)).transpose() +
154  p1.segment<3>(9) * basisDx(p1.segment<2>(0)).transpose());
155  Eigen::MatrixXd t_curl(3, polynomial_degree_plus_one_squared);
156  t_curl = (1.0 / kappa2) *
157  (-p2.segment<3>(6) * basisDy(p2.segment<2>(0)).transpose() +
158  p2.segment<3>(9) * basisDx(p2.segment<2>(0)).transpose());
159  // inner product of surface curls of any two basis functions
160  for (int j = 0; j < polynomial_degree_plus_one_squared; ++j)
161  for (int i = 0; i < polynomial_degree_plus_one_squared; ++i)
162  (*intval)(j * polynomial_degree_plus_one_squared + i) +=
163  w * s_curl.col(i).dot(t_curl.col(j));
164  }

183  void addScaledVectorBasisInteraction(
184  Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>* intval, Scalar w,
185  const Eigen::Vector2d& s, const Eigen::Vector2d& t,
186  const Eigen::Vector3d x_f_dx, const Eigen::Vector3d x_f_dy,
187  const Eigen::Vector3d y_f_dx, const Eigen::Vector3d y_f_dy) const {
188  auto basis_interaction = basisInteraction(s, t);
189  intval->block(0, 0, polynomial_degree_plus_one_squared,
190  polynomial_degree_plus_one_squared) +=
191  w * x_f_dx.dot(y_f_dx) * basis_interaction;
192  intval->block(0, polynomial_degree_plus_one_squared,
193  polynomial_degree_plus_one_squared,
194  polynomial_degree_plus_one_squared) +=
195  w * x_f_dx.dot(y_f_dy) * basis_interaction;
196  intval->block(polynomial_degree_plus_one_squared, 0,
197  polynomial_degree_plus_one_squared,
198  polynomial_degree_plus_one_squared) +=
199  w * x_f_dy.dot(y_f_dx) * basis_interaction;
200  intval->block(polynomial_degree_plus_one_squared,
201  polynomial_degree_plus_one_squared,
202  polynomial_degree_plus_one_squared,
203  polynomial_degree_plus_one_squared) +=
204  w * x_f_dy.dot(y_f_dy) * basis_interaction;
205  }
