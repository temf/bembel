34 class MaxwellSingleLayerOperator
35  : public LinearOperatorBase<MaxwellSingleLayerOperator> {
36  // implementation of the kernel evaluation, which may be based on the
37  // information available from the superSpace
38  public:
39  MaxwellSingleLayerOperator() {}
40  template <class T>
41  void evaluateIntegrand_impl(
42  const T &super_space, const SurfacePoint &p1, const SurfacePoint &p2,
43  Eigen::Matrix<
44  typename LinearOperatorTraits<MaxwellSingleLayerOperator>::Scalar,
45  Eigen::Dynamic, Eigen::Dynamic> *intval) const {
46  auto polynomial_degree = super_space.get_polynomial_degree();
47  auto polynomial_degree_plus_one_squared =
48  (polynomial_degree + 1) * (polynomial_degree + 1);
49 
50  // get evaluation points on unit square
51  auto s = p1.segment<2>(0);
52  auto t = p2.segment<2>(0);
53 
54  // get quadrature weights
55  auto ws = p1(2);
56  auto wt = p2(2);
57 
58  // get points on geometry and tangential derivatives
59  auto x_f = p1.segment<3>(3);
60  auto x_f_dx = p1.segment<3>(6);
61  auto x_f_dy = p1.segment<3>(9);
62  auto y_f = p2.segment<3>(3);
63  auto y_f_dx = p2.segment<3>(6);
64  auto y_f_dy = p2.segment<3>(9);
65 
66  // compute surface measures from tangential derivatives
67  auto h = 1. / (1 << super_space.get_refinement_level()); // h = 1 ./ (2^M)
68 
69  // integrand without basis functions
70  auto kernel_evaluation = evaluateKernel(x_f, y_f) * ws * wt;
71  auto integrand_vector = kernel_evaluation;
72  auto integrand_divergence = -kernel_evaluation / wavenumber2_ / h / h;
73 
74  // vector part: mulitply shape functions with integrand and add to buffer
75  super_space.addScaledVectorBasisInteraction(intval, integrand_vector, s, t,
76  x_f_dx, x_f_dy, y_f_dx, y_f_dy);
77 
78  // divergence part: multiply shape functions with integrand and add to
79  // buffer
80  super_space.addScaledVectorBasisDivergenceInteraction(
81  intval, integrand_divergence, s, t);
82 
83  return;
84  }

86  Eigen::Matrix<std::complex<double>, 4, 4> evaluateFMMInterpolation_impl(
87  const SurfacePoint &p1, const SurfacePoint &p2) const {
88  // get evaluation points on unit square
89  auto s = p1.segment<2>(0);
90  auto t = p2.segment<2>(0);
91 
92  // get points on geometry and tangential derivatives
93  auto x_f = p1.segment<3>(3);
94  auto x_f_dx = p1.segment<3>(6);
95  auto x_f_dy = p1.segment<3>(9);
96  auto y_f = p2.segment<3>(3);
97  auto y_f_dx = p2.segment<3>(6);
98  auto y_f_dy = p2.segment<3>(9);
99 
100  // evaluate kernel
101  auto kernel = evaluateKernel(x_f, y_f);
102 
103  // interpolation
104  Eigen::Matrix<std::complex<double>, 4, 4> intval;
105  intval.setZero();
106  intval(0, 0) = kernel * x_f_dx.dot(y_f_dx);
107  intval(0, 2) = kernel * x_f_dx.dot(y_f_dy);
108  intval(2, 0) = kernel * x_f_dy.dot(y_f_dx);
109  intval(2, 2) = kernel * x_f_dy.dot(y_f_dy);
110  intval(1, 1) = -kernel / wavenumber2_;
111  intval(1, 3) = -kernel / wavenumber2_;
112  intval(3, 1) = -kernel / wavenumber2_;
113  intval(3, 3) = -kernel / wavenumber2_;
114 
115  return intval;
116  }
117 
121  std::complex<double> evaluateKernel(const Eigen::Vector3d &x,
122  const Eigen::Vector3d &y) const {
123  auto r = (x - y).norm();
124  return std::exp(-std::complex<double>(0., 1.) * wavenumber_ * r) / 4. /
125  BEMBEL_PI / r;
126  }
128  // setters
130  void set_wavenumber(std::complex<double> wavenumber) {
131  wavenumber_ = wavenumber;
132  wavenumber2_ = wavenumber_ * wavenumber_;
133  }
135  // getters
137  std::complex<double> get_wavenumber() { return wavenumber_; }
138 
139  private:
140  std::complex<double> wavenumber_;
141  std::complex<double> wavenumber2_;
142 };

149 template <typename InterpolationPoints>
150 struct H2Multipole::Moment2D<InterpolationPoints, MaxwellSingleLayerOperator> {
151  static std::vector<Eigen::MatrixXd> compute2DMoment(
152  const SuperSpace<MaxwellSingleLayerOperator> &super_space,
153  const int cluster_level, const int cluster_refinements,
154  const int number_of_points) {
155  Eigen::MatrixXd moment = moment2DComputer<
156  Moment1D<InterpolationPoints, MaxwellSingleLayerOperator>,
157  Moment1D<InterpolationPoints, MaxwellSingleLayerOperator>>(
158  super_space, cluster_level, cluster_refinements, number_of_points);
159  Eigen::MatrixXd moment_dx = moment2DComputer<
160  Moment1DDerivative<InterpolationPoints, MaxwellSingleLayerOperator>,
161  Moment1D<InterpolationPoints, MaxwellSingleLayerOperator>>(
162  super_space, cluster_level, cluster_refinements, number_of_points);
163  Eigen::MatrixXd moment_dy = moment2DComputer<
164  Moment1D<InterpolationPoints, MaxwellSingleLayerOperator>,
165  Moment1DDerivative<InterpolationPoints, MaxwellSingleLayerOperator>>(
166  super_space, cluster_level, cluster_refinements, number_of_points);
167 
168  Eigen::MatrixXd moment1(moment.rows() + moment_dx.rows(), moment.cols());
169  moment1 << moment, moment_dx;
170  Eigen::MatrixXd moment2(moment.rows() + moment_dy.rows(), moment.cols());
171  moment2 << moment, moment_dy;
172 
173  std::vector<Eigen::MatrixXd> vector_of_moments;
174  vector_of_moments.push_back(moment1);
175  vector_of_moments.push_back(moment2);
176 
177  return vector_of_moments;
178  }
179 };
