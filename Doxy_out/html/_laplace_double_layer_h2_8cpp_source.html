27 int main() {
28  using namespace Bembel;
29  using namespace Eigen;
30 
31  Bembel::IO::Stopwatch sw;
32 
33  int polynomial_degree_max = 2;
34  int refinement_level_max = 3;
35 
36  // Load geometry from file "sphere.dat", which must be placed in the same
37  // directory as the executable
38  Geometry geometry("sphere.dat");
39 
40  // Define evaluation points for potential field, a tensor product grid of
41  // 7*7*7 points in [-.1,.1]^3
42  MatrixXd gridpoints = Util::makeTensorProductGrid(
43  VectorXd::LinSpaced(10, -.25, .25), VectorXd::LinSpaced(10, -.25, .25),
44  VectorXd::LinSpaced(10, -.25, .25));
45 
46  // Define analytical solution using lambda function, in this case a harmonic
47  // function, see Data.hpp
48  std::function<double(Vector3d)> fun = [](Vector3d in) {
49  return Data::HarmonicFunction(in);
50  };
51 
52  std::cout << "\n" << std::string(60, '=') << "\n";
53  // Iterate over polynomial degree
54  for (int polynomial_degree = 0; polynomial_degree < polynomial_degree_max + 1;
55  ++polynomial_degree) {
56  VectorXd error(refinement_level_max + 1);
57  // Iterate over refinement levels
58  for (int refinement_level = 0; refinement_level < refinement_level_max + 1;
59  ++refinement_level) {
60  sw.tic();
61  std::cout << "Degree " << polynomial_degree << " Level "
62  << refinement_level << "\t\t";
63  // Build ansatz space
64  AnsatzSpace<LaplaceDoubleLayerOperator> ansatz_space_helm(
65  geometry, refinement_level, polynomial_degree);
66  AnsatzSpace<MassMatrixScalarDisc> ansatz_space_mass(
67  geometry, refinement_level, polynomial_degree);
68 
69  // Set up load vector
70  DiscreteLinearForm<DirichletTrace<double>, LaplaceDoubleLayerOperator>
71  disc_lf(ansatz_space_helm);
72  disc_lf.get_linear_form().set_function(fun);
73  disc_lf.compute();
74 
75  // Set up and compute discrete operator
76  DiscreteOperator<H2Matrix<double>, LaplaceDoubleLayerOperator>
77  disc_op_double(ansatz_space_helm);
78  disc_op_double.compute();
79  const H2Matrix<double> &K =
80  disc_op_double.get_discrete_operator();
81 
82  // Assemble mass matrix and system matrix
83  DiscreteLocalOperator<MassMatrixScalarDisc> disc_op_mass(
84  ansatz_space_mass);
85  disc_op_mass.compute();
86  SparseMatrix<double> M = disc_op_mass.get_discrete_operator();
87  auto system_matrix = -0.5 * M + K; // important: do NOT change auto!
88 
89  // solve system
90  GMRES<typeof(system_matrix), IdentityPreconditioner> gmres;
91  gmres.compute(system_matrix);
92  VectorXd rho = gmres.solve(disc_lf.get_discrete_linear_form());
93 
94  // evaluate potential
95  DiscretePotential<LaplaceDoubleLayerPotential<LaplaceDoubleLayerOperator>,
96  LaplaceDoubleLayerOperator>
97  disc_pot(ansatz_space_helm);
98  disc_pot.set_cauchy_data(rho);
99  auto pot = disc_pot.evaluate(gridpoints);
100 
101  error(refinement_level) = maxPointwiseError<double>(pot, gridpoints, fun);
102  std::cout << " time " << std::setprecision(4) << sw.toc() << "s\t\t";
103  std::cout << error(refinement_level) << std::endl;
104  }
105 
106  // estimate rate of convergence and check whether it is at least 90% of the
107  // expected value
108  assert(
109  checkRateOfConvergence(error.tail(3), 2 * polynomial_degree + 2, 0.9));
110 
111  std::cout << std::endl;
112  }
113  std::cout << std::string(60, '=') << std::endl;
114 
115  return 0;
116 }
