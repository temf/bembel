33  void init_DiscretePotential(const AnsatzSpace<LinOp> &ansatz_space) {
34  ansatz_space_ = ansatz_space;
35  fun_ev_ = FunctionEvaluator<LinOp>(ansatz_space_);
39  deg_ = ansatz_space_.get_polynomial_degree() + 1;
40  return;
41  }
43  // compute
45  static_assert(
46  getFunctionSpaceOutputDimension<LinearOperatorTraits<LinOp>::Form>() ==
47  PotentialTraits<Derived>::OutputSpaceDimension,
48  "Dimension mismatch in potential evaluation");
49  Eigen::Matrix<typename PotentialReturnScalar<
50  typename LinearOperatorTraits<LinOp>::Scalar,
51  typename PotentialTraits<Derived>::Scalar>::Scalar,
52  Eigen::Dynamic, PotentialTraits<Derived>::OutputSpaceDimension>
53  evaluate(const Eigen::Matrix<double, Eigen::Dynamic, 3> &points) {
54  auto FunctionSpaceVectorDimension =
55  getFunctionSpaceVectorDimension<LinearOperatorTraits<LinOp>::Form>();
56  auto OutputDimension = PotentialTraits<Derived>::OutputSpaceDimension;
57 
58  GaussSquare<Constants::maximum_quadrature_degree> GS;
59  auto Q = GS[deg_];
60 
61  auto super_space = ansatz_space_.get_superspace();
62 
63  const ElementTree &element_tree = super_space.get_mesh().get_element_tree();
64  auto number_of_elements = element_tree.get_number_of_elements();
65 
66  auto polynomial_degree = super_space.get_polynomial_degree();
67  auto polynomial_degree_plus_one_squared =
68  (polynomial_degree + 1) * (polynomial_degree + 1);
69 
70  Eigen::Matrix<typename PotentialReturnScalar<
71  typename LinearOperatorTraits<LinOp>::Scalar,
72  typename PotentialTraits<Derived>::Scalar>::Scalar,
73  Eigen::Dynamic,
74  PotentialTraits<Derived>::OutputSpaceDimension>
75  potential;
76  potential.resize(points.rows(),
77  PotentialTraits<Derived>::OutputSpaceDimension);
78  potential.setZero();
79 
80 #pragma omp parallel
81  {
82  Eigen::Matrix<typename PotentialReturnScalar<
83  typename LinearOperatorTraits<LinOp>::Scalar,
84  typename PotentialTraits<Derived>::Scalar>::Scalar,
85  Eigen::Dynamic,
86  PotentialTraits<Derived>::OutputSpaceDimension>
87  my_potential;
88  my_potential.resize(points.rows(),
89  PotentialTraits<Derived>::OutputSpaceDimension);
90  my_potential.setZero();
91  for (auto element = element_tree.cpbegin();
92  element != element_tree.cpend(); ++element) {
93 #pragma omp single nowait
94  {
95  SurfacePoint qp;
96  for (auto j = 0; j < Q.w_.size(); ++j) {
97  super_space.map2surface(
98  *element, Q.xi_.col(j),
99  element->get_h() * element->get_h() * Q.w_(j), &qp);
100  for (auto i = 0; i < points.rows(); ++i) {
101  my_potential.row(i) += pot_.evaluateIntegrand_impl(
102  fun_ev_, *element, points.row(i), qp);
103  }
104  }
105  }
106  }
107 #pragma omp critical
108  potential += my_potential;
109  }
110  return potential;
111  }
