19 template <typename Matrix1, typename Matrix2>
20 void testOpPlus(const Matrix1& m1, const Matrix2& m2) {
21  typedef typename Eigen::Matrix<typename Matrix1::Scalar, Eigen::Dynamic, 1>
22  VectorType;
23  assert(m1.rows() == m2.rows());
24  assert(m1.cols() == m2.cols());
25  VectorType a = VectorType::Random(m1.cols());
26  VectorType b1 = m1 * a;
27  VectorType b2 = m2 * a;
28  VectorType ref = b1.eval() + b2.eval();
29  auto matrix_op = m1 + m2;
30  VectorType test = matrix_op * a;
31  assert((ref - test).norm() < 1e-14);
32 }
33 template <typename Matrix1, typename Matrix2>
34 void testOpMinus(const Matrix1& m1, const Matrix2& m2) {
35  typedef typename Eigen::Matrix<typename Matrix1::Scalar, Eigen::Dynamic, 1>
36  VectorType;
37  assert(m1.rows() == m2.rows());
38  assert(m1.cols() == m2.cols());
39  VectorType a = VectorType::Random(m1.cols());
40  VectorType b1 = m1 * a;
41  VectorType b2 = m2 * a;
42  VectorType ref = b1.eval() - b2.eval();
43  auto matrix_op = m1 - m2;
44  VectorType test = matrix_op * a;
45  assert((ref - test).norm() < 1e-14);
46 }

52  int polynomial_degree = 1;

54 
55  // Load geometry from file "sphere.dat", which must be placed in the same
56  // directory as the executable
57  Geometry geometry("sphere.dat");
58 
59  // define types
60  typedef HelmholtzSingleLayerOperator Operator;
61  typedef typename LinearOperatorTraits<Operator>::Scalar Scalar;
62  typedef typename LinearOperatorTraits<Operator>::EigenType VectorType;
63  typedef H2Matrix<Scalar> H2MatrixType;
64  typedef Matrix<Scalar, Dynamic, Dynamic> DenseMatrixType;
65  typedef SparseMatrix<Scalar> SparseMatrixType;
66 
67  // Build ansatz space
68  AnsatzSpace<Operator> ansatz_space(geometry, refinement_level,
69  polynomial_degree);
70 
71  // Set up and compute discrete operator
72  DiscreteOperator<H2MatrixType, Operator> disc_op(ansatz_space);
73  disc_op.compute();
74  const H2MatrixType& H2 = disc_op.get_discrete_operator();
75 
76  { // interaction with itself
77  testOpPlus(H2, H2);
78  testOpPlus(H2, H2 + H2);
79  testOpPlus(H2 + H2, H2);
80  testOpPlus(H2, H2 - H2);
81  testOpPlus(H2 - H2, H2);
82  testOpMinus(H2, H2);
83  testOpMinus(H2, H2 + H2);
84  testOpMinus(H2 + H2, H2);
85  testOpMinus(H2, H2 - H2);
86  testOpMinus(H2 - H2, H2);
87  }
88  { // interaction with dense matrix
89  DenseMatrixType S(H2.rows(), H2.cols());
90  S.setIdentity();
91 
92  testOpPlus(S, H2);
93  testOpPlus(S, H2 + H2);
94  testOpPlus(S, H2 - H2);

96  testOpPlus(H2 + H2, S);

104  }
105  { // interaction with sparse matrix

111  testOpPlus(S, H2 - H2);

118  testOpMinus(H2, S);

120  testOpMinus(H2 - H2, S);
121  }
122  { // interaction with scalar
123  Scalar b = 3.;

130  }
