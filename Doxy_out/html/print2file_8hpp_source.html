75 template <typename Scalar>
76 int print2m(const std::string &fileName, const std::string &varName,
77  const Eigen::SparseMatrix<Scalar> &var,
78  const std::string &writeMode) {
79  Eigen::VectorXd rowInd(var.nonZeros());
80  Eigen::VectorXd colInd(var.nonZeros());
81  Eigen::VectorXd value(var.nonZeros());
82  unsigned int j = 0;
83  for (auto i = 0; i < var.outerSize(); i++)
84  for (typename Eigen::SparseMatrix<Scalar>::InnerIterator it(var, i); it;
85  ++it) {
86  rowInd(j) = it.row() + 1;
87  colInd(j) = it.col() + 1;
88  value(j) = it.value();
89  ++j;
90  }
91  print2m(fileName, "rows_" + varName, rowInd, writeMode);
92  print2m(fileName, "cols_" + varName, colInd, "a");
93  print2m(fileName, "values_" + varName, value, "a");
94  std::ofstream myfile;
95  // if flag is set to w, a new file is created, otherwise the new matrix
96  // is just appended
97  myfile.open(fileName, std::ios_base::app);
98  myfile << varName << " = sparse("
99  << "rows_" + varName << ","
100  << "cols_" + varName << ","
101  << "values_" + varName << ");\n";
102  myfile.close();
103 
104  return 0;
105 }

107 template <typename Derived>
108 int print2bin(const std::string &fileName,
109  const Eigen::MatrixBase<Derived> &var) {
110  std::ofstream myfile;
111  int rows = 0;
112  int cols = 0;
113  int IsRowMajor = 0;
114  int dataSize = 0;
115 
116  IsRowMajor = var.IsRowMajor;
117  dataSize = sizeof(typename Derived::Scalar);
118 
119  myfile.open(fileName, std::ios::out | std::ios::binary | std::ios::trunc);
120  // write data size and row major flag
121  myfile.write(reinterpret_cast<const char *>(&dataSize), sizeof(int));
122  myfile.write(reinterpret_cast<const char *>(&IsRowMajor), sizeof(int));
123  rows = var.rows();
124  cols = var.cols();
125  // write rows and cols of the matrix
126  myfile.write((const char *)&(rows), sizeof(int));
127  myfile.write((const char *)&(cols), sizeof(int));
128  std::cout << "writing binary file" << std::endl;
129  std::cout << "rows: " << rows << " cols: " << cols
130  << " dataSize: " << dataSize << " IsRowMajor: " << IsRowMajor
131  << std::endl;
132  if (IsRowMajor) {
133  Eigen::Matrix<typename Derived::Scalar, Eigen::Dynamic, Eigen::Dynamic,
134  Eigen::RowMajor>
135  tmp = var;
136 
137  myfile.write(reinterpret_cast<const char *>(tmp.data()),
138  rows * cols * dataSize);
139 
140  } else {
141  Eigen::Matrix<typename Derived::Scalar, Eigen::Dynamic, Eigen::Dynamic,
142  Eigen::ColMajor>
143  tmp = var;
144 
145  myfile.write(reinterpret_cast<const char *>(tmp.data()),
146  rows * cols * dataSize);
147  }
148 
149  myfile.close();
150 
151  return 0;
152 }

154 template <typename Derived>
155 int bin2Mat(const std::string &fileName,
156  Eigen::MatrixBase<Derived> *targetMat) {
157  std::ifstream myfile;
158  int rows = 0;
159  int cols = 0;
160  int IsRowMajor = 0;
161  int dataSize = 0;
162  typename Derived::Scalar *data = NULL;
163  Eigen::Matrix<typename Derived::Scalar, Eigen::Dynamic, Eigen::Dynamic>
164  returnMat;
165 
166  myfile.open(fileName, std::ios::in | std::ios::binary);
167 
168  myfile.read(reinterpret_cast<char *>(&dataSize), sizeof(int));
169  myfile.read(reinterpret_cast<char *>(&IsRowMajor), sizeof(int));
170  myfile.read(reinterpret_cast<char *>(&rows), sizeof(int));
171  myfile.read(reinterpret_cast<char *>(&cols), sizeof(int));
172  std::cout << "reading binary file" << std::endl;
173  std::cout << "rows: " << rows << " cols: " << cols
174  << " dataSize: " << dataSize << " IsRowMajor: " << IsRowMajor
175  << std::endl;
176  if (dataSize != sizeof(typename Derived::Scalar)) {
177  std::cout << "mismatch in data size of target and input file size"
178  << std::endl;
179  return 1;
180  }
181 
182  data = new typename Derived::Scalar[rows * cols];
183 
184  myfile.read((char *)data, rows * cols * dataSize);
185 
186  myfile.close();
187 
188  if (IsRowMajor)
189  returnMat =
190  Eigen::Map<Eigen::Matrix<typename Derived::Scalar, Eigen::Dynamic,
191  Eigen::Dynamic, Eigen::RowMajor> >(data, rows,
192  cols);
193  else
194  returnMat =
195  Eigen::Map<Eigen::Matrix<typename Derived::Scalar, Eigen::Dynamic,
196  Eigen::Dynamic, Eigen::ColMajor> >(data, rows,
197  cols);
198 
199  *targetMat = returnMat;
200 
201  delete[] data;
202  return 0;
203 }
