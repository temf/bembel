40  void init_ElementTree(const Geometry &g, unsigned int max_level) {
41  // initialise the data fields
42  geometry_ = g.get_geometry_ptr();
43  max_level_ = max_level;
44  number_of_patches_ = geometry_->size();
45  // create the patches and set up the topology
46  {
47  std::vector<Eigen::Vector3d> uniquePts;
48  std::vector<ElementTreeNode *> patches;
49  Eigen::Vector3d v;
50  number_of_points_ = 0;
51  number_of_elements_ = number_of_patches_;
52  ElementTreeNode &root = root_;
53  root.sons_.resize(number_of_patches_);
54  for (auto i = 0; i < number_of_patches_; ++i) {
55  root.sons_[i].adjcents_ = std::vector<ElementTreeNode *>(4, nullptr);
56  root.sons_[i].vertices_.resize(4);
57  root.sons_[i].id_ = i;
58  root.sons_[i].level_ = 0;
59  root.sons_[i].patch_ = i;
60  // add linked list structure to the panels
61  if (i == 0) {
62  root.sons_[0].prev_ = nullptr;
63  pfirst_ = std::addressof(root.sons_[0]);
64  } else {
65  root.sons_[i].prev_ = std::addressof(root.sons_[i - 1]);
66  }
67  if (i == number_of_patches_ - 1) {
68  root.sons_[i].next_ = nullptr;
69  plast_ = std::addressof(root.sons_[i]);
70  } else {
71  root.sons_[i].next_ = std::addressof(root.sons_[i + 1]);
72  }
73  // get images of the four corners of the unit square under the
74  // diffeomorphism geo[i]
75  for (auto j = 0; j < 4; ++j) {
76  v = (*geometry_)[i].eval(Constants::corners[0][j],
77  Constants::corners[1][j]);
78  unsigned int index = 0;
79  for (; index < uniquePts.size(); ++index)
80  if ((uniquePts[index] - v).norm() < Constants::pt_comp_tolerance)
81  break;
82  if (index != uniquePts.size()) {
83  root.sons_[i].vertices_[j] = index;
84  } else {
85  uniquePts.push_back(v);
86  root.sons_[i].vertices_[j] = number_of_points_;
87  ++number_of_points_;
88  }
89  }
90  patches.push_back(std::addressof(root.sons_[i]));
91  }
92  updateTopology(patches);
93  }
94  for (auto i = 0; i < max_level; ++i) refineUniformly();
95  return;
96  }

118  Eigen::MatrixXd generatePointList(Eigen::VectorXi *idct = nullptr) const {
119  Eigen::MatrixXd pts(3, number_of_points_);
120  if (idct != nullptr) {
121  idct->resize(number_of_points_);
122  idct->setZero();
123  }
124  for (auto it = pbegin(); it != pend(); ++it) {
125  double h = double(1) / double(1 << it->level_);
126  pts.col(it->vertices_[0]) =
127  (*geometry_)[it->patch_].eval(it->llc_(0), it->llc_(1));
128  pts.col(it->vertices_[1]) =
129  (*geometry_)[it->patch_].eval(it->llc_(0) + h, it->llc_(1));
130  pts.col(it->vertices_[2]) =
131  (*geometry_)[it->patch_].eval(it->llc_(0) + h, it->llc_(1) + h);
132  pts.col(it->vertices_[3]) =
133  (*geometry_)[it->patch_].eval(it->llc_(0), it->llc_(1) + h);
134  if (idct != nullptr) {
135  ++((*idct)(it->vertices_[0]));
136  ++((*idct)(it->vertices_[1]));
137  ++((*idct)(it->vertices_[2]));
138  ++((*idct)(it->vertices_[3]));
139  }
140  }
141  return pts;
142  }

190  void computeElementEnclosings_recursion(ElementTreeNode &el,
191  const Eigen::MatrixXd &P) {
192  Eigen::Vector3d mp1, mp2;
193  double r1, r2;
194  // compute point list
195  if (!el.sons_.size()) {
196  // assign enclosing balls to leafs
197  computeEnclosingBall(&mp1, &r1, P.col(el.vertices_[0]), 0,
198  P.col(el.vertices_[2]), 0);
199  computeEnclosingBall(&mp2, &r2, P.col(el.vertices_[1]), 0,
200  P.col(el.vertices_[3]), 0);
201  computeEnclosingBall(&(el.midpoint_), &(el.radius_), mp1, r1, mp2, r2);
202  } else {
203  // handle the four(!!!) children
204  for (auto i = 0; i < 4; ++i)
205  computeElementEnclosings_recursion(el.sons_[i], P);
206  // assign enclosing balls to fathers bottom up
207  computeEnclosingBall(&mp1, &r1, el.sons_[0].midpoint_,
208  el.sons_[0].radius_, el.sons_[2].midpoint_,
209  el.sons_[2].radius_);
210  computeEnclosingBall(&mp2, &r2, el.sons_[1].midpoint_,
211  el.sons_[1].radius_, el.sons_[3].midpoint_,
212  el.sons_[3].radius_);
213  computeEnclosingBall(&(el.midpoint_), &(el.radius_), mp1, r1, mp2, r2);
214  }
215  return;
216  }

294  static void computeEnclosingBall(Eigen::Vector3d *mp, double *r,
295  const Eigen::Vector3d &mp1, double r1,
296  const Eigen::Vector3d &mp2, double r2) {
297  // compute distance vector of the two spheres
298  auto z = mp1 - mp2;
299  auto norm = (mp1 - mp2).norm();
301  if (norm + r2 <= r1) {
302  *mp = mp1;
303  *r = r1;
305  } else if (norm + r1 <= r2) {
306  *mp = mp2;
307  *r = r2;
309  } else {
310  *mp = 0.5 * (mp1 + mp2 + (r1 - r2) / norm * z);
311  *r = 0.5 * (r1 + r2 + norm);
312  *r = 0.5 * (r1 + r2 + norm);
313  }
314  return;
315  }
320  std::vector<std::array<int, 4>> patchTopologyInfo() const {
321  std::vector<std::array<int, 4>> retval;
322  for (auto it = root_.sons_.begin(); it != root_.sons_.end(); ++it) {
323  for (auto j = 0; j < 4; ++j) {
324  // do we have a neighbour?
325  if (it->adjcents_[j] != nullptr) {
326  const ElementTreeNode &cur_neighbour = *(it->adjcents_[j]);
327  // add the edge only if it->id_ < neighbour->id_
328  // otherwise the edge has already been added
329  if (it->id_ < cur_neighbour.id_) {
330  int k = 0;
331  for (; k < 4; ++k)
332  if (cur_neighbour.adjcents_[k] != nullptr &&
333  cur_neighbour.adjcents_[k]->id_ == it->id_)
334  break;
335  retval.push_back({it->id_, cur_neighbour.id_, j, k});
336  }
337  } else {
338  retval.push_back({it->id_, -1, j, -1});
339  }
340  }
341  }
342  return retval;
343  }
344  // The ordering of elements in the element tree does not correspond to the
345  // element order underlying the coefficient vector. This reordering can be

349  std::vector<int> computeReorderingVector() const {
350  std::vector<int> out(number_of_elements_);
351  for (auto it = pbegin(); it != pend(); ++it) {
352  const double h = it->get_h();
353  const Eigen::Vector2d ref_mid =
354  it->llc_ + Eigen::Vector2d(.5 * h, .5 * h);
355  const int x_idx = std::floor(ref_mid(0) / h);
356  const int y_idx = std::floor(ref_mid(1) / h);
357  const int num_in_one_dir = 1 << max_level_;
358  const int num_per_patch = num_in_one_dir * num_in_one_dir;
359  const int tp_idx =
360  it->patch_ * num_per_patch + y_idx * num_in_one_dir + x_idx;
361  out[tp_idx] = it->id_;
362  }
363  return out;

389  void updateTopology(const std::vector<ElementTreeNode *> &elements) {
390  std::map<std::array<int, 2>, ElementTreeNode *> edges;
391  std::array<int, 2> e1, e2;
392  // generate edge list for all elements in question
393  for (auto i = 0; i < elements.size(); ++i)
394  for (auto j = 0; j < 4; ++j) {
395  // compute a unique id for each edge
396  const int v1 = elements[i]->vertices_[j];
397  const int v2 = elements[i]->vertices_[(j + 1) % 4];
398  e1 = {v1 < v2 ? v1 : v2, v1 < v2 ? v2 : v1};
399  // perferm a look up if the edge is already existing.
400  auto it = edges.find(e1);
401  // if so, identify the two neighbours
402  if (it != edges.end()) {
403  elements[i]->adjcents_[j] = it->second;
404  // now find the edge also in the patch that added it to the set
405  for (auto k = 0; k < 4; ++k) {
406  const int v3 = it->second->vertices_[k];
407  const int v4 = it->second->vertices_[(k + 1) % 4];
408  e2 = {v3 < v4 ? v3 : v4, v3 < v4 ? v4 : v3};
409  if (e1 == e2) {
410  it->second->adjcents_[k] = elements[i];
411  break;
412  }
413  }
414  // otherwise add the edge to the list
415  } else {
416  edges.insert(std::make_pair(e1, elements[i]));
417  }
418  }
419  return;
420  }
422  void refineLeaf(ElementTreeNode &cur_el) {
423  // check if we have actually a panel
424  if (cur_el.sons_.size()) return;
425  std::vector<int> ptIds(5);
426  std::vector<int> refNeighbours(4);
427  std::vector<ElementTreeNode *> elements;
428  // determine position of p with respect to its neighbours
429  for (auto i = 0; i < 4; ++i) {
430  // is there a neighbour?
431  if (cur_el.adjcents_[i] != nullptr) {
432  for (auto j = 0; j < 4; ++j)
433  if (cur_el.adjcents_[i]->adjcents_[j] == std::addressof(cur_el)) {
434  refNeighbours[i] = j;
435  break;
436  }
437  } else {
438  refNeighbours[i] = -1;
439  }
440  }
441  // determine new points
442  for (auto i = 0; i < 4; ++i) {
443  // is there a neighbour?
444  if (cur_el.adjcents_[i] != nullptr) {
445  ElementTreeNode &ref_cur_neighbour = *(cur_el.adjcents_[i]);
446  // is the neighbour already refined?
447  if (ref_cur_neighbour.sons_.size()) {
448  // this is the midpoint of the shared edge
449  ptIds[i] = ref_cur_neighbour.sons_[refNeighbours[i]]
450  .vertices_[(refNeighbours[i] + 1) % 4];
451  // these are the two elements adjacent to the edge
452  elements.push_back(
453  std::addressof(ref_cur_neighbour.sons_[refNeighbours[i]]));
454  elements.push_back(std::addressof(
455  ref_cur_neighbour.sons_[(refNeighbours[i] + 1) % 4]));
456  } else {
457  // otherwise add the point id to the tree
458  ptIds[i] = number_of_points_;
459  ++number_of_points_;
460  }
461  } else {
462  // otherwise add the point id to the tree
463  ptIds[i] = number_of_points_;
464  ++number_of_points_;
465  }
466  }
467  // add midpoint of the current element, which is always a new point
468  ptIds[4] = number_of_points_;
469  ++number_of_points_;
470  // set up new elements
471  cur_el.sons_.resize(4);
472  number_of_elements_ += 3;
473  max_level_ =
474  max_level_ < cur_el.level_ + 1 ? cur_el.level_ + 1 : max_level_;
475  // auto it = leafs_.find(compute_global_id(cur_el));
476  // if (it != leafs_.end()) leafs_.erase(it);
477  for (auto i = 0; i < 4; ++i) {
478  // add linked list structure to the panels
480  if (i == 0) {
481  cur_el.sons_[i].prev_ = cur_el.prev_;
482  if (cur_el.prev_ != nullptr)
483  (cur_el.prev_)->next_ = std::addressof(cur_el.sons_[i]);
484  cur_el.prev_ = nullptr;
485  if (std::addressof(cur_el) == pfirst_)
486  pfirst_ = std::addressof(cur_el.sons_[i]);
487 
488  } else {
489  cur_el.sons_[i].prev_ = std::addressof(cur_el.sons_[i - 1]);
490  }
491  if (i == 3) {
492  cur_el.sons_[i].next_ = cur_el.next_;
493  if (cur_el.next_ != nullptr)
494  (cur_el.next_)->prev_ = std::addressof(cur_el.sons_[i]);
495  cur_el.next_ = nullptr;
496  if (std::addressof(cur_el) == plast_)
497  plast_ = std::addressof(cur_el.sons_[i]);
498  } else {
499  cur_el.sons_[i].next_ = std::addressof(cur_el.sons_[i + 1]);
500  }
502  cur_el.sons_[i].patch_ = cur_el.patch_;
503  cur_el.sons_[i].level_ = cur_el.level_ + 1;
504  cur_el.sons_[i].id_ = 4 * cur_el.id_ + i;
505  cur_el.sons_[i].adjcents_ = std::vector<ElementTreeNode *>(4, nullptr);
506  cur_el.sons_[i].llc_(0) =
507  cur_el.llc_(0) + Constants::llcs[0][i] / double(1 << cur_el.level_);
508  cur_el.sons_[i].llc_(1) =
509  cur_el.llc_(1) + Constants::llcs[1][i] / double(1 << cur_el.level_);
510  elements.push_back(std::addressof(cur_el.sons_[i]));
511  }
512  // set vertices
513  // first element
514  cur_el.sons_[0].vertices_.push_back(cur_el.vertices_[0]);
515  cur_el.sons_[0].vertices_.push_back(ptIds[0]);
516  cur_el.sons_[0].vertices_.push_back(ptIds[4]);
517  cur_el.sons_[0].vertices_.push_back(ptIds[3]);
518  // second element
519  cur_el.sons_[1].vertices_.push_back(ptIds[0]);
520  cur_el.sons_[1].vertices_.push_back(cur_el.vertices_[1]);
521  cur_el.sons_[1].vertices_.push_back(ptIds[1]);
522  cur_el.sons_[1].vertices_.push_back(ptIds[4]);
523  // third element
524  cur_el.sons_[2].vertices_.push_back(ptIds[4]);
525  cur_el.sons_[2].vertices_.push_back(ptIds[1]);
526  cur_el.sons_[2].vertices_.push_back(cur_el.vertices_[2]);
527  cur_el.sons_[2].vertices_.push_back(ptIds[2]);
528  // fourth element
529  cur_el.sons_[3].vertices_.push_back(ptIds[3]);
530  cur_el.sons_[3].vertices_.push_back(ptIds[4]);
531  cur_el.sons_[3].vertices_.push_back(ptIds[2]);
532  cur_el.sons_[3].vertices_.push_back(cur_el.vertices_[3]);
533  // fix adjecency relations
534  updateTopology(elements);
535 
536  return;
537  }
