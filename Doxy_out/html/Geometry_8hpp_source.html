45  inline void init_Geometry(const std::string &filename) {
46  // Note that the Shredder is required. The order of ansatz functions allows
47  // to be chosen higher than the smoothness of the NÃ™RBS mappings. Thus, we
48  // need to shredder the geometry mappings to have Bezier patches. You can
49  // achieve the higher regularity by changing coefficients in the projector.
50  auto tmp = Bembel::PatchShredder(Bembel::LoadGeometryFile(filename));
51  geometry_ = std::make_shared<PatchVector>();
52  *geometry_ = tmp;
53  }
