44 template <int N, int P>
45 inline constexpr double Bernstein(double evaluation_point) noexcept {
46  return Binomial<N, P>::value * BernsteinX<N>(evaluation_point) *
47  BernsteinX<P - N>(1. - evaluation_point);
48 }
52 template <typename T, int N, int P>
53 class HiddenBernsteinClass {
54  public:
55  static inline T EvalCoefs(T *in, double evaluation_point) noexcept {
56  return in[N] * Bernstein<N, P>(evaluation_point) +
57  HiddenBernsteinClass<T, N - 1, P>::EvalCoefs(in, evaluation_point);
58  }
59  static inline T EvalDerCoefs(T *in, double evaluation_point) noexcept {
60  return (
61  (in[N + 1] - in[N]) * Bernstein<N, P>(evaluation_point) +
62  HiddenBernsteinClass<T, N - 1, P>::EvalDerCoefs(in, evaluation_point));
63  }
64  static inline void EvalBasisPEQ(T *in, double evaluation_point) noexcept {
65  in[N] += Bernstein<N, P>(evaluation_point);
66  HiddenBernsteinClass<T, N - 1, P>::EvalBasisPEQ(in, evaluation_point);
67  return;
68  }
69  static inline void EvalDerBasisPEQ(T *in, double evaluation_point) noexcept {
70  in[N] += (P + 1) * (Bernstein<N - 1, P>(evaluation_point) -
71  Bernstein<N, P>(evaluation_point));
72  HiddenBernsteinClass<T, N - 1, P>::EvalDerBasisPEQ(in, evaluation_point);
73  return;
74  }
75  static inline void EvalBasis(T *in, double evaluation_point) noexcept {
76  in[N] = Bernstein<N, P>(evaluation_point);
77  HiddenBernsteinClass<T, N - 1, P>::EvalBasis(in, evaluation_point);
78  return;
79  }
80  static inline void EvalDerBasis(T *in, double evaluation_point) noexcept {
81  in[N] = (P + 1) * (Bernstein<N - 1, P>(evaluation_point) -
82  Bernstein<N, P>(evaluation_point));
83  HiddenBernsteinClass<T, N - 1, P>::EvalDerBasis(in, evaluation_point);
84  return;
85  }
86 };

93  }
