49  template<class T>
50  void evaluateIntegrand_impl(const T &super_space, const SurfacePoint &p1,
51  const SurfacePoint &p2,
52  Eigen::Matrix<
53  typename LinearOperatorTraits<HomogenisedLaplaceSingleLayerOperator
54  >::Scalar, Eigen::Dynamic, Eigen::Dynamic> *intval) const {
55  auto polynomial_degree = super_space.get_polynomial_degree();
56  auto polynomial_degree_plus_one_squared = (polynomial_degree + 1)
57  * (polynomial_degree + 1);
58 
59  // get evaluation points on unit square
60  auto s = p1.segment < 2 > (0);
61  auto t = p2.segment < 2 > (0);
62 
63  // get quadrature weights
64  auto ws = p1(2);
65  auto wt = p2(2);
66 
67  // get points on geometry and tangential derivatives
68  auto x_f = p1.segment < 3 > (3);
69  auto x_f_dx = p1.segment < 3 > (6);
70  auto x_f_dy = p1.segment < 3 > (9);
71  auto y_f = p2.segment < 3 > (3);
72  auto y_f_dx = p2.segment < 3 > (6);
73  auto y_f_dy = p2.segment < 3 > (9);
74 
75  // compute surface measures from tangential derivatives
76  auto x_kappa = x_f_dx.cross(x_f_dy).norm();
77  auto y_kappa = y_f_dx.cross(y_f_dy).norm();
78 
79  // integrand without basis functions
80  auto integrand = evaluateKernel(x_f, y_f) * x_kappa * y_kappa * ws * wt;
81 
82  // multiply basis functions with integrand and add to intval, this is an
83  // efficient implementation of
84  // (*intval) += super_space.basisInteraction(s, t)
85  // * evaluateKernel(x_f, y_f) * x_kappa * y_kappa * ws * wt;
86  super_space.addScaledBasisInteraction(intval, integrand, s, t);
87 
88  return;
89  }

91  Eigen::Matrix<double, 1, 1> evaluateFMMInterpolation_impl(
92  const SurfacePoint &p1, const SurfacePoint &p2) const {
93  // get evaluation points on unit square
94  auto s = p1.segment < 2 > (0);
95  auto t = p2.segment < 2 > (0);
96 
97  // get points on geometry and tangential derivatives
98  auto x_f = p1.segment < 3 > (3);
99  auto x_f_dx = p1.segment < 3 > (6);
100  auto x_f_dy = p1.segment < 3 > (9);
101  auto y_f = p2.segment < 3 > (3);
102  auto y_f_dx = p2.segment < 3 > (6);
103  auto y_f_dy = p2.segment < 3 > (9);
104 
105  // compute surface measures from tangential derivatives
106  auto x_kappa = x_f_dx.cross(x_f_dy).norm();
107  auto y_kappa = y_f_dx.cross(y_f_dy).norm();
108 
109  // interpolation
110  Eigen::Matrix<double, 1, 1> intval;
111  intval(0) = evaluateKernel(x_f, y_f) * x_kappa * y_kappa;
112 
113  return intval;
114  }
