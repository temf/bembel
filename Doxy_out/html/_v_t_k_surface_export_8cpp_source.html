22 int main() {
23  using namespace Bembel;
24  Geometry geo("sphere.dat");
25 
26  // The refinement level for the visualization is independent of that of the
27  // simulation since one might consider to visualize a coarse discretisation on
28  // a smooth geometry.
29  const int refinement_level = 4;
30 
31  // The VTKwriter sets up initial geomety information.
32  VTKSurfaceExport writer(geo, refinement_level);
33 
34  // Now we can add user defined data. There are different options. Since you
35  // might consider to visualize a solution of a computation, we will set up a
36  // "fake" ansatz-space and showcase that.
37  AnsatzSpace<FakeDivergenceConformingOperator> aspace(geo, 2, 1, 1);
38  Eigen::VectorXd coefficients(aspace.get_number_of_dofs());
39  for (int j = 0; j < aspace.get_number_of_dofs(); ++j) {
40  coefficients(j) = (j % 10) * 0.1;
41  }
42  FunctionEvaluator<FakeDivergenceConformingOperator> evaluator(aspace);
43  evaluator.set_function(coefficients);
44 
45  // One can use either one of the following formats:
46  // std::function<double(int, Eigen::Vector2d)>& fun)
47  // std::function<Eigen::Vector3d(int, Eigen::Vector2d)>
48  // std::function<double(Eigen::Vector3d)>
49  // std::function<Eigen::Vector3d(Eigen::Vector3d)>
50  std::function<Eigen::Vector3d(int, const Eigen::Vector2d&)> fun1 =
51  [&](int patch_number, const Eigen::Vector2d& reference_domain_point) {
52  return evaluator.evaluateOnPatch(patch_number, reference_domain_point);
53  };
54 
55  std::function<double(const Eigen::Vector3d&)> fun2 =
56  [](const Eigen::Vector3d& point_in_space) { return point_in_space(0); };
57 
58  // With the help of these functions, the data sets are generated by the
59  // VTKWriter
60  writer.addDataSet("Vector_Field", fun1);
61  writer.addDataSet("X-Value", fun2);
62 
63  // Finally, we print to file.
64  writer.writeToFile("example.vtp");
65  return 0;
66 }
