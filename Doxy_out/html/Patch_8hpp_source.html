34  inline void init_Patch(const std::vector<Eigen::Matrix<double, -1, -1>> &xyzw,
35  const std::vector<double> &x_knots,
36  const std::vector<double> &y_knots) {
37  assert(xyzw.size() == 4);
38  const int xyzw_cols = xyzw[0].cols();
39  const int xyzw_rows = xyzw[0].rows();
40  unique_knots_x_ = Spl::ExtractUniqueKnotVector(x_knots);
41  unique_knots_y_ = Spl::ExtractUniqueKnotVector(y_knots);
42  const int xnumpatch = unique_knots_x_.size() - 1;
43  const int ynumpatch = unique_knots_y_.size() - 1;
44  polynomial_degree_x_ = x_knots.size() - xyzw_cols;
45  polynomial_degree_y_ = y_knots.size() - xyzw_rows;
46  data_.resize(4 * (polynomial_degree_x_ * xnumpatch * polynomial_degree_y_ *
47  ynumpatch));
48  {
49  // Since its only for initialization, I do not care about speed.
50  // Here I look weather the data given is already in bezier form.
51  if (unique_knots_x_.size() == 2 && unique_knots_y_.size() == 2) {
52  for (int i = 0; i < 4; i++) {
53  Eigen::Matrix<double, -1, 1> tmp = Spl::Unroll(xyzw[i]);
54  for (int j = 0; j < tmp.rows(); j++) data_[j * 4 + i] = (tmp[j]);
55  }
56  } else {
57  // If not, I construct the dynamic projection (i.e. solve
58  // systems for
59  // the coeffs) and project to the superspace.
60 
61  Eigen::SparseMatrix<double> phi = Spl::MakeProjection(
62  x_knots, y_knots, unique_knots_x_, unique_knots_y_,
63  polynomial_degree_x_, polynomial_degree_y_);
64 
65  for (int i = 0; i < 4; i++) {
66  Eigen::Matrix<double, -1, 1> tmp =
67  Spl::Unroll(xyzw[i]).transpose() * phi.transpose();
68 
69  for (int j = 0; j < tmp.rows(); j++) data_[j * 4 + i] = (tmp[j]);
70  }
71  }
72  }
73 
74  return;
75  }

80  Eigen::Vector3d eval(const Eigen::Vector2d &reference_point) const {
81  const int x_location =
82  Spl::FindLocationInKnotVector(reference_point(0), unique_knots_x_);
83  const int y_location =
84  Spl::FindLocationInKnotVector(reference_point(1), unique_knots_y_);
85  const int numy = (unique_knots_y_.size() - 1) * polynomial_degree_y_;
86  const double scaledx =
87  Spl::Rescale(reference_point(0), unique_knots_x_[x_location],
88  unique_knots_x_[x_location + 1]);
89  const double scaledy =
90  Spl::Rescale(reference_point(1), unique_knots_y_[y_location],
91  unique_knots_y_[y_location + 1]);
92 
93  double *xbasis = new double[polynomial_degree_x_];
94  double *ybasis = new double[polynomial_degree_y_];
95 
96  Bembel::Basis::ShapeFunctionHandler::evalBasis(polynomial_degree_x_ - 1,
97  xbasis, scaledx);
98  Bembel::Basis::ShapeFunctionHandler::evalBasis(polynomial_degree_y_ - 1,
99  ybasis, scaledy);
100 
101  double tmp[4] = {0., 0., 0., 0.};
102 
103  for (int i = 0; i < polynomial_degree_x_; i++) {
104  for (int j = 0; j < polynomial_degree_y_; j++) {
105  const double tpbasisval = xbasis[i] * ybasis[j];
106  const int accs = 4 * (numy * (polynomial_degree_x_ * x_location + i) +
107  polynomial_degree_y_ * y_location + j);
108 #pragma omp simd
109  for (int k = 0; k < 4; k++) tmp[k] += data_[accs + k] * tpbasisval;
110  }
111  }
112 
113  delete[] xbasis;
114  delete[] ybasis;
115 
116  Eigen::Vector3d out(tmp[0], tmp[1], tmp[2]);
117  // Rescaling by the NRBS weight, i.e. projection to 3D from 4D hom
118 
119  return out / tmp[3];
120  }

122  Eigen::Matrix<double, 3, 2> evalJacobian(
123  const Eigen::Vector2d &reference_point) const {
124  const int x_location =
125  Spl::FindLocationInKnotVector(reference_point(0), unique_knots_x_);
126  const int y_location =
127  Spl::FindLocationInKnotVector(reference_point(1), unique_knots_y_);
128  const int numy = (unique_knots_y_.size() - 1) * polynomial_degree_y_;
129  const double scaledx =
130  Spl::Rescale(reference_point(0), unique_knots_x_[x_location],
131  unique_knots_x_[x_location + 1]);
132  const double scaledy =
133  Spl::Rescale(reference_point(1), unique_knots_y_[y_location],
134  unique_knots_y_[y_location + 1]);
135 
136  double *xbasis = new double[polynomial_degree_x_];
137  double *ybasis = new double[polynomial_degree_y_];
138  double *xbasisD = new double[polynomial_degree_x_];
139  double *ybasisD = new double[polynomial_degree_y_];
140 
141  Bembel::Basis::ShapeFunctionHandler::evalBasis(polynomial_degree_x_ - 1,
142  xbasis, scaledx);
143  Bembel::Basis::ShapeFunctionHandler::evalBasis(polynomial_degree_y_ - 1,
144  ybasis, scaledy);
145  Bembel::Basis::ShapeFunctionHandler::evalDerBasis(polynomial_degree_x_ - 1,
146  xbasisD, scaledx);
147  Bembel::Basis::ShapeFunctionHandler::evalDerBasis(polynomial_degree_y_ - 1,
148  ybasisD, scaledy);
149 
150  double tmp[4] = {0., 0., 0., 0.};
151  double tmpDx[4] = {0., 0., 0., 0.};
152  double tmpDy[4] = {0., 0., 0., 0.};
153 
154  for (int i = 0; i < polynomial_degree_x_; i++) {
155  for (int j = 0; j < polynomial_degree_y_; j++) {
156  const double tpbasisval = xbasis[i] * ybasis[j];
157  const double tpbasisvalDx = xbasisD[i] * ybasis[j];
158  const double tpbasisvalDy = xbasis[i] * ybasisD[j];
159  const int accs = 4 * (numy * (polynomial_degree_x_ * x_location + i) +
160  polynomial_degree_y_ * y_location + j);
161 
162  // Here I add up the values of the basis functions in the dc
163  // basis
164 #pragma omp simd
165  for (int k = 0; k < 4; k++) {
166  tmp[k] += data_[accs + k] * tpbasisval;
167  tmpDx[k] += data_[accs + k] * tpbasisvalDx;
168  tmpDy[k] += data_[accs + k] * tpbasisvalDy;
169  }
170  }
171  }
172 
173  delete[] xbasis;
174  delete[] ybasis;
175  delete[] xbasisD;
176  delete[] ybasisD;
177 
178  Eigen::Matrix<double, 3, 2> out;
179 
180  // Eigen::Vector3d out;
181 
182  double bot = 1. / (tmp[3] * tmp[3]);
183 
184 #pragma omp simd
185  for (int k = 0; k < 3; k++) {
186  out(k, 0) = (tmpDx[k] * tmp[3] - tmp[k] * tmpDx[3]) * bot;
187  out(k, 1) = (tmpDy[k] * tmp[3] - tmp[k] * tmpDy[3]) * bot;
188  }
189 
190  return out;
191  }

212  void updateSurfacePoint(SurfacePoint *srf_pt,
213  const Eigen::Vector2d &ref_pt, double w,
214  const Eigen::Vector2d &xi) const {
215  const int x_location =
216  Spl::FindLocationInKnotVector(ref_pt(0), unique_knots_x_);
217  const int y_location =
218  Spl::FindLocationInKnotVector(ref_pt(1), unique_knots_y_);
219  const int numy = (unique_knots_y_.size() - 1) * polynomial_degree_y_;
220  const double scaledx = Spl::Rescale(ref_pt(0), unique_knots_x_[x_location],
221  unique_knots_x_[x_location + 1]);
222  const double scaledy = Spl::Rescale(ref_pt(1), unique_knots_y_[y_location],
223  unique_knots_y_[y_location + 1]);
224 
225  // TODO(Felix) Do not use variable-length arrays in accordance to Google
226  // Style guide
227  double *buffer =
228  new double[2 * (polynomial_degree_x_ + polynomial_degree_y_) + 12];
229  for (int i = 0; i < 12; ++i) buffer[i] = 0;
230 
231  double *tmp = buffer;
232  double *tmpDx = tmp + 4;
233  double *tmpDy = tmpDx + 4;
234  double *xbasis = tmpDy + 4;
235  double *ybasis = xbasis + polynomial_degree_x_;
236  double *xbasisD = ybasis + polynomial_degree_y_;
237  double *ybasisD = xbasisD + polynomial_degree_x_;
238 
239  Bembel::Basis::ShapeFunctionHandler::evalBasis(polynomial_degree_x_ - 1,
240  xbasis, scaledx);
241  Bembel::Basis::ShapeFunctionHandler::evalBasis(polynomial_degree_y_ - 1,
242  ybasis, scaledy);
243  Bembel::Basis::ShapeFunctionHandler::evalDerBasis(polynomial_degree_x_ - 1,
244  xbasisD, scaledx);
245  Bembel::Basis::ShapeFunctionHandler::evalDerBasis(polynomial_degree_y_ - 1,
246  ybasisD, scaledy);
247 
248  for (int i = 0; i < polynomial_degree_x_; ++i) {
249  for (int j = 0; j < polynomial_degree_y_; ++j) {
250  const double tpbasisval = xbasis[i] * ybasis[j];
251  const double tpbasisvalDx = xbasisD[i] * ybasis[j];
252  const double tpbasisvalDy = xbasis[i] * ybasisD[j];
253  const int accs = 4 * (numy * (polynomial_degree_x_ * x_location + i) +
254  polynomial_degree_y_ * y_location + j);
255 
256  // Here I add up the values of the basis functions in the dc
257  // basis
258  for (int k = 0; k < 4; ++k) {
259  tmp[k] += data_[accs + k] * tpbasisval;
260  tmpDx[k] += data_[accs + k] * tpbasisvalDx;
261  tmpDy[k] += data_[accs + k] * tpbasisvalDy;
262  }
263  }
264  }
265 
266  const double bot = 1. / tmp[3];
267  const double botsqr = bot * bot;
268 
269  (*srf_pt)(0) = xi(0);
270  (*srf_pt)(1) = xi(1);
271  (*srf_pt)(2) = w;
272  (*srf_pt)(3) = tmp[0] * bot;
273  (*srf_pt)(4) = tmp[1] * bot;
274  (*srf_pt)(5) = tmp[2] * bot;
275  (*srf_pt)(6) = (tmpDx[0] * tmp[3] - tmp[0] * tmpDx[3]) * botsqr;
276  (*srf_pt)(7) = (tmpDx[1] * tmp[3] - tmp[1] * tmpDx[3]) * botsqr;
277  (*srf_pt)(8) = (tmpDx[2] * tmp[3] - tmp[2] * tmpDx[3]) * botsqr;
278  (*srf_pt)(9) = (tmpDy[0] * tmp[3] - tmp[0] * tmpDy[3]) * botsqr;
279  (*srf_pt)(10) = (tmpDy[1] * tmp[3] - tmp[1] * tmpDy[3]) * botsqr;
280  (*srf_pt)(11) = (tmpDy[2] * tmp[3] - tmp[2] * tmpDy[3]) * botsqr;
281  delete[] buffer;
282  return;
283  }

298 inline std::vector<Patch> PatchShredder(const Patch &patch) noexcept {
299  // Already a Bezier patch
300  if (patch.unique_knots_y_.size() == 2 && patch.unique_knots_x_.size() == 2) {
301  return {patch};
302  }
303 
304  // number of subpatches in x and y directions
305  const int xchips = patch.unique_knots_x_.size() - 1;
306  const int ychips = patch.unique_knots_y_.size() - 1;
307 
308  const int xp = patch.polynomial_degree_x_;
309  const int yp = patch.polynomial_degree_y_;
310  const int numy = ychips * yp;
311 
312  std::vector<Patch> out(xchips * ychips);
313 
314  for (int ix = 0; ix < xchips; ix++) {
315  for (int iy = 0; iy < ychips; iy++) {
316  const int index = ix * ychips + iy;
317 
318  out[index].unique_knots_x_ = {0, 1};
319  out[index].unique_knots_y_ = {0, 1};
320  out[index].polynomial_degree_x_ = xp;
321  out[index].polynomial_degree_y_ = yp;
322  out[index].data_.reserve(xp * yp * 4);
323  }
324  }
325 
326  for (int ix = 0; ix < xchips; ix++) {
327  for (int iy = 0; iy < ychips; iy++) {
328  const int index = ix * ychips + iy;
329  for (int jx = 0; jx < xp; jx++) {
330  for (int jy = 0; jy < yp; jy++) {
331  const int accs = 4 * (numy * (xp * ix + jx) + yp * iy + jy);
332  for (int k = 0; k < 4; k++) {
333  out[index].data_.push_back(patch.data_[accs + k]);
334  }
335  }
336  }
337  }
338  }
339 
340  return out;
341 }

343 // Shredds a whole vector of Patches
344 inline std::vector<Patch> PatchShredder(
345  const std::vector<Patch> &patches) noexcept {
346  std::vector<Patch> out;
347  const int input_size = patches.size();
348 
349  for (int i = 0; i < input_size; i++) {
350  std::vector<Patch> tmp = PatchShredder(patches[i]);
351 
352  const int tmp_size = tmp.size();
353 
354  for (int j = 0; j < tmp_size; j++) out.push_back(tmp[j]);
355  }
356 
357  out.shrink_to_fit();
358 
359  return out;
360 }
